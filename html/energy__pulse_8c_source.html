125static int setup_pulse_gpio(const struct gpio_dt_spec *gpio_spec,
126 struct gpio_callback *gpio_cb)
127{
128 int err;
129
130 if (!device_is_ready(gpio_spec->port)) {
131 LOG_ERR("Pulse GPIO device not ready");
132 return -ENODEV;
133 }
134
135 err = gpio_pin_configure_dt(gpio_spec, GPIO_INPUT | GPIO_PULL_UP);
136 if (err) {
137 LOG_ERR("Failed to configure pulse GPIO pin, error: %d", err);
138 return err;
139 }
140
141 err = gpio_add_callback(gpio_spec->port, gpio_cb);
142 if (err) {
143 LOG_ERR("Failed to add GPIO callback, error: %d", err);
144 return err;
145 }
146
147 err = gpio_pin_interrupt_configure_dt(gpio_spec, GPIO_INT_EDGE_TO_ACTIVE);
148 if (err) {
149 LOG_ERR("Failed to configure GPIO interrupt, error: %d", err);
150 return err;
151 }
152
153 return 0;
154}

194static void energy_pulse_msg_publish(void)
195{
196 atomic_val_t pulse_count_ch1, pulse_count_ch2;
197 int32_t pulse_energy_ch1_wh, pulse_energy_ch2_wh;
198 int64_t timestamp_ms = 0;
199 int err;
200
201 /* Atomically read pulse counts for both channels */
202 pulse_count_ch1 = atomic_get(&pulse_energy_ch1);
203 pulse_count_ch2 = atomic_get(&pulse_energy_ch2);
204
205 /* Convert pulse counts to Wh using configured conversion factor (pure int32_t!) */
206 pulse_energy_ch1_wh = (int32_t)pulse_count_ch1 * ENERGY_PULSE_WH_PER_PULSE;
207 pulse_energy_ch2_wh = (int32_t)pulse_count_ch2 * ENERGY_PULSE_WH_PER_PULSE;
208
209 /* Get the current timestamp in milliseconds */
210 err = date_time_now(&timestamp_ms);
211 if (err) {
212 LOG_ERR("Failed to get current date/time, error: %d", err);
213 }
214
215 /* Prepare the message struct with the latest data */
216 struct energy_pulse_msg msg = {
217 .timestamp = timestamp_ms,
218 .pulse_energy_ch1_wh = pulse_energy_ch1_wh,
219 .pulse_energy_ch2_wh = pulse_energy_ch2_wh,
220 };
221
222 /* Publish the message to the ZBUS channel */
223 err = zbus_chan_pub(&ENERGY_PULSE_CHAN, &msg, K_SECONDS(1));
224 if (err) {
225 LOG_ERR("Failed to publish energy pulse message, error: %d", err);
226 }
227
228 LOG_DBG("Published energy pulse: timestamp=%lld, ch1=%d Wh, ch2=%d Wh",
229 msg.timestamp, msg.pulse_energy_ch1_wh, msg.pulse_energy_ch2_wh);
230}

248static void pulse_gpio_callback(const struct device *dev,
249 struct gpio_callback *cb, uint32_t pins)
250{
251 ARG_UNUSED(dev);
252 ARG_UNUSED(cb);
253
254 if (pins & BIT(pulse_ch1_gpio.pin)) {
255 atomic_inc(&pulse_energy_ch1);
256 atomic_val_t new_count = atomic_get(&pulse_energy_ch1);
257 LOG_DBG("Energy pulse detected! CH1 new count: %lu", new_count);
258
259 /* Mark that we have new data since last save */
260 atomic_set(&has_new_pulses_since_save, 1);
261 }
262
263 if (pins & BIT(pulse_ch2_gpio.pin)) {
264 atomic_inc(&pulse_energy_ch2);
265 atomic_val_t new_count = atomic_get(&pulse_energy_ch2);
266 LOG_DBG("Energy pulse detected! CH2 new count: %lu", new_count);
267
268 /* Mark that we have new data since last save */
269 atomic_set(&has_new_pulses_since_save, 1);
270 }
271}

310static int energy_pulse_settings_set(const char *name, size_t len,
311 settings_read_cb read_cb, void *cb_arg)
312{
313 const char *next;
314 int rc;
315
316 /* Validate data size */
317 if (len != sizeof(atomic_val_t)) {
318 LOG_ERR("Invalid data size for '%s': %zu (expected %zu)", name, len,
319 sizeof(atomic_val_t));
320 return -EINVAL;
321 }
322
323 /* Check for channel 1 key */
324 if (settings_name_steq(name, "ch1", &next) && !next) {
325 atomic_val_t value;
326
327 rc = read_cb(cb_arg, &value, sizeof(value));
328 if (rc >= 0) {
329 atomic_set(&pulse_energy_ch1, value);
330 LOG_DBG("Loaded channel 1: %lu pulses", value);
331 return 0;
332 }
333
334 LOG_ERR("Failed to read channel 1 data, error: %d", rc);
335 return rc;
336 }
337
338 /* Check for channel 2 key */
339 if (settings_name_steq(name, "ch2", &next) && !next) {
340 atomic_val_t value;
341
342 rc = read_cb(cb_arg, &value, sizeof(value));
343 if (rc >= 0) {
344 atomic_set(&pulse_energy_ch2, value);
345 LOG_DBG("Loaded channel 2: %lu pulses", value);
346 return 0;
347 }
348
349 LOG_ERR("Failed to read channel 2 data, error: %d", rc);
350 return rc;
351 }
352
353 /* Unknown key */
354 LOG_WRN("Unknown settings key: '%s'", name);
355 return -ENOENT;
356}

394static int save_energy_counters(void)
395{
396 /* Check if we have new data since last save */
397 if (atomic_get(&has_new_pulses_since_save) == 0) {
398 LOG_DBG("No new pulses since last save, skipping");
399 return 0;
400 }
401
402 atomic_val_t pulse_count_ch1, pulse_count_ch2;
403 int err_ch1, err_ch2;
404
405 /* Atomically read pulse counts */
406 pulse_count_ch1 = atomic_get(&pulse_energy_ch1);
407 pulse_count_ch2 = atomic_get(&pulse_energy_ch2);
408
409 /* Save both channels independently */
410 err_ch1 = save_energy_channel(ENERGY_PULSE_CH1_KEY, pulse_count_ch1);
411 err_ch2 = save_energy_channel(ENERGY_PULSE_CH2_KEY, pulse_count_ch2);
412
413 /* Return error if either channel failed to save */
414 if (err_ch1 || err_ch2) {
415 return (err_ch1 != 0) ? err_ch1 : err_ch2;
416 }
417
418 /* Reset the new data flag since we just saved */
419 atomic_set(&has_new_pulses_since_save, 0);
420 LOG_INF("Energy counters saved to persistent storage");
421
422 return 0;
423}

446 if (err) {
447 LOG_ERR("Failed to load settings, error: %d", err);

449 }
450

453
454 LOG_INF("Energy pulse initialized - ch1=%lu pulses, ch2=%lu pulses",
455 atomic_get(&pulse_energy_ch1), atomic_get(&pulse_energy_ch2));
456
457 return 0;

471static void energy_pulse_thread(void)
472{
473 const uint32_t wdt_timeout_ms =
474 CONFIG_APP_ENERGY_PULSE_MODULE_WATCHDOG_TIMEOUT_SECONDS *
475 MSEC_PER_SEC;
476 const uint32_t msg_processing_ms =
477 CONFIG_APP_ENERGY_PULSE_MODULE_MSG_PROCESSING_TIME_SECONDS *
478 MSEC_PER_SEC;
479 const k_timeout_t zbus_wait_ms = K_MSEC(wdt_timeout_ms - msg_processing_ms);
480 const struct zbus_channel *chan;
481 uint8_t msg_buf[MAX_MSG_SIZE];
482 int err;
483 int task_wdt_id;
484
485 LOG_DBG("Energy pulse started - WDT: %u s, Processing: %u s, Wait: %u s",
486 wdt_timeout_ms / MSEC_PER_SEC, msg_processing_ms / MSEC_PER_SEC,
487 (wdt_timeout_ms - msg_processing_ms) / MSEC_PER_SEC);
488
489 /* Setup watchdog */
490 task_wdt_id = task_wdt_util_setup(wdt_timeout_ms, task_wdt_callback,
491 (void *)k_current_get());
492 if (task_wdt_id < 0) {
493 LOG_ERR("Failed to setup watchdog");
494 return;
495 }
496
497 /* Load energy counters from settings */
498 err = energy_pulse_load_settings();
499 if (err) {
500 LOG_ERR("Failed to load energy pulse settings, using defaults");
501 }
502
503 /* Initialize GPIO for pulse input */
504 err = energy_pulse_gpio_init();
505 if (err) {
506 LOG_ERR("Failed to initialize GPIO for pulse input, error: %d", err);
507 }
508
509 /* Attach ZBUS subscriber to thread */
510 zbus_obs_attach_to_thread(&energy_pulse_subscriber);
511
512 /* Main loop */
513 while (true) {
514 err = task_wdt_feed(task_wdt_id);
515 if (err) {
516 LOG_ERR("Failed to feed task watchdog, error: %d", err);
517 return;
518 }
519
520 /* Wait for message (MSG_SUBSCRIBER gets message copy directly) */
521 err = zbus_sub_wait_msg(&energy_pulse_subscriber, &chan, msg_buf,
522 zbus_wait_ms);
523 if (err == -ENOMSG) {
524 continue;
525 } else if (err) {
526 LOG_ERR("zbus_sub_wait_msg, error: %d", err);
527 return;
528 }
529
530 /* Process message based on channel */
531 if (chan == &DATA_PUBLISH_REQUEST_CHAN) {
532 const struct timing_event_msg *publish_req =
533 MSG_TO_TIMING_EVENT_MSG_PTR(msg_buf);
534 LOG_DBG("Received publish request with timestamp: %lld ms",
535 publish_req->timestamp);
536
537 /* Publish current energy pulse data */
538 energy_pulse_msg_publish();
539
540 /* Save energy counters to flash for persistence */
541 err = save_energy_counters();
542 if (err) {
543 LOG_WRN("Failed to persist energy counters, error: %d", err);
544 }
545 } else if (chan == &REBOOT_CHAN) {
546 const struct reboot_msg *reboot = MSG_TO_REBOOT_MSG_PTR(msg_buf);
547 LOG_DBG("Received reboot message with type: %d", reboot->type);
548
549 /* Save energy counters before shutdown/reboot */
550 if (reboot->type == SHUTDOWN_REQ) {
551 LOG_INF("Shutdown requested, saving energy counters...");
552 err = save_energy_counters();
553 if (err) {
554 LOG_ERR("Failed to save energy counters during shutdown, error: %d",
555 err);
556 }
557
558 /* Send shutdown acknowledgement */
559 shutdown_acknowledge_publish(&energy_pulse_subscriber);
560 }
561 }
562 }
563}
