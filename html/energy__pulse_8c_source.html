122static int setup_pulse_gpio(const struct gpio_dt_spec *gpio_spec,
123 struct gpio_callback *gpio_cb)
124{
125 int err;
126
127 if (!device_is_ready(gpio_spec->port)) {
128 LOG_ERR("Pulse GPIO device not ready");
129 return -ENODEV;
130 }
131
132 err = gpio_pin_configure_dt(gpio_spec, GPIO_INPUT | GPIO_PULL_UP);
133 if (err) {
134 LOG_ERR("Failed to configure pulse GPIO pin, error: %d", err);
135 return err;
136 }
137
138 err = gpio_add_callback(gpio_spec->port, gpio_cb);
139 if (err) {
140 LOG_ERR("Failed to add GPIO callback, error: %d", err);
141 return err;
142 }
143
144 err = gpio_pin_interrupt_configure_dt(gpio_spec, GPIO_INT_EDGE_TO_ACTIVE);
145 if (err) {
146 LOG_ERR("Failed to configure GPIO interrupt, error: %d", err);
147 return err;
148 }
149
150 return 0;
151}

191static void energy_pulse_msg_publish(void)
192{
193 atomic_val_t pulse_count_ch1, pulse_count_ch2;
194 int32_t pulse_energy_ch1_wh, pulse_energy_ch2_wh;
195 int64_t timestamp_ms = 0;
196 int err;
197
198 /* Atomically read pulse counts for both channels */
199 pulse_count_ch1 = atomic_get(&pulse_energy_ch1);
200 pulse_count_ch2 = atomic_get(&pulse_energy_ch2);
201
202 /* Convert pulse counts to Wh using configured conversion factor (pure int32_t!) */
203 pulse_energy_ch1_wh = (int32_t)pulse_count_ch1 * ENERGY_PULSE_WH_PER_PULSE;
204 pulse_energy_ch2_wh = (int32_t)pulse_count_ch2 * ENERGY_PULSE_WH_PER_PULSE;
205
206 /* Get the current uptime timestamp in milliseconds */
207 int64_t uptime_ms = k_uptime_get();
208
209 /* Prepare the message struct with the latest data */
210 struct energy_pulse_msg msg = {
211 .uptime_ms = uptime_ms,
212 .pulse_energy_ch1_wh = pulse_energy_ch1_wh,
213 .pulse_energy_ch2_wh = pulse_energy_ch2_wh,
214 };
215
216 /* Publish the message to the ZBUS channel */
217 err = zbus_chan_pub(&ENERGY_PULSE_CHAN, &msg, K_SECONDS(1));
218 if (err) {
219 LOG_ERR("Failed to publish energy pulse message, error: %d", err);
220 }
221
222 LOG_DBG("Published energy pulse: uptime=%lld ms, ch1=%d Wh, ch2=%d Wh",
223 msg.uptime_ms, msg.pulse_energy_ch1_wh, msg.pulse_energy_ch2_wh);
224}

242static void pulse_gpio_callback(const struct device *dev,
243 struct gpio_callback *cb, uint32_t pins)
244{
245 ARG_UNUSED(dev);
246 ARG_UNUSED(cb);
247
248 if (pins & BIT(pulse_ch1_gpio.pin)) {
249 atomic_inc(&pulse_energy_ch1);
250 atomic_val_t new_count = atomic_get(&pulse_energy_ch1);
251 LOG_DBG("Energy pulse detected! CH1 new count: %lu", new_count);
252
253 /* Mark that we have new data since last save */
254 atomic_set(&has_new_pulses_since_save, 1);
255 }
256
257 if (pins & BIT(pulse_ch2_gpio.pin)) {
258 atomic_inc(&pulse_energy_ch2);
259 atomic_val_t new_count = atomic_get(&pulse_energy_ch2);
260 LOG_DBG("Energy pulse detected! CH2 new count: %lu", new_count);
261
262 /* Mark that we have new data since last save */
263 atomic_set(&has_new_pulses_since_save, 1);
264 }
265}

304static int energy_pulse_settings_set(const char *name, size_t len,
305 settings_read_cb read_cb, void *cb_arg)
306{
307 const char *next;
308 int rc;
309
310 /* Validate data size */
311 if (len != sizeof(atomic_val_t)) {
312 LOG_ERR("Invalid data size for '%s': %zu (expected %zu)", name, len,
313 sizeof(atomic_val_t));
314 return -EINVAL;
315 }
316
317 /* Check for channel 1 key */
318 if (settings_name_steq(name, "ch1", &next) && !next) {
319 atomic_val_t value;
320
321 rc = read_cb(cb_arg, &value, sizeof(value));
322 if (rc >= 0) {
323 atomic_set(&pulse_energy_ch1, value);
324 LOG_DBG("Loaded channel 1: %lu pulses", value);
325 return 0;
326 }
327
328 LOG_ERR("Failed to read channel 1 data, error: %d", rc);
329 return rc;
330 }
331
332 /* Check for channel 2 key */
333 if (settings_name_steq(name, "ch2", &next) && !next) {
334 atomic_val_t value;
335
336 rc = read_cb(cb_arg, &value, sizeof(value));
337 if (rc >= 0) {
338 atomic_set(&pulse_energy_ch2, value);
339 LOG_DBG("Loaded channel 2: %lu pulses", value);
340 return 0;
341 }
342
343 LOG_ERR("Failed to read channel 2 data, error: %d", rc);
344 return rc;
345 }
346
347 /* Unknown key */
348 LOG_WRN("Unknown settings key: '%s'", name);
349 return -ENOENT;
350}

388static int save_energy_counters(void)
389{
390 /* Check if we have new data since last save */
391 if (atomic_get(&has_new_pulses_since_save) == 0) {
392 LOG_DBG("No new pulses since last save, skipping");
393 return 0;
394 }
395
396 atomic_val_t pulse_count_ch1, pulse_count_ch2;
397 int err_ch1, err_ch2;
398
399 /* Atomically read pulse counts */
400 pulse_count_ch1 = atomic_get(&pulse_energy_ch1);
401 pulse_count_ch2 = atomic_get(&pulse_energy_ch2);
402
403 /* Save both channels independently */
404 err_ch1 = save_energy_channel(ENERGY_PULSE_CH1_KEY, pulse_count_ch1);
405 err_ch2 = save_energy_channel(ENERGY_PULSE_CH2_KEY, pulse_count_ch2);
406
407 /* Return error if either channel failed to save */
408 if (err_ch1 || err_ch2) {
409 return (err_ch1 != 0) ? err_ch1 : err_ch2;
410 }
411
412 /* Reset the new data flag since we just saved */
413 atomic_set(&has_new_pulses_since_save, 0);
414 LOG_INF("Energy counters saved to persistent storage");
415
416 return 0;
417}

465static void energy_pulse_thread(void)
466{
467 const uint32_t wdt_timeout_ms =
468 CONFIG_APP_ENERGY_PULSE_MODULE_WATCHDOG_TIMEOUT_SECONDS *
469 MSEC_PER_SEC;
470 const uint32_t msg_processing_ms =
471 CONFIG_APP_ENERGY_PULSE_MODULE_MSG_PROCESSING_TIME_SECONDS *
472 MSEC_PER_SEC;
473 const k_timeout_t zbus_wait_ms = K_MSEC(wdt_timeout_ms - msg_processing_ms);
474 const struct zbus_channel *chan;
475 uint8_t msg_buf[MAX_MSG_SIZE];
476 int err;
477 int task_wdt_id;
478
479 LOG_DBG("Energy pulse started - WDT: %u s, Processing: %u s, Wait: %u s",
480 wdt_timeout_ms / MSEC_PER_SEC, msg_processing_ms / MSEC_PER_SEC,
481 (wdt_timeout_ms - msg_processing_ms) / MSEC_PER_SEC);
482
483 /* Setup watchdog */
484 task_wdt_id = task_wdt_util_setup(wdt_timeout_ms, task_wdt_callback,
485 (void *)k_current_get());
486 if (task_wdt_id < 0) {
487 LOG_ERR("Failed to setup watchdog");
488 return;
489 }
490
491 /* Load energy counters from settings */
492 err = energy_pulse_load_settings();
493 if (err) {
494 LOG_ERR("Failed to load energy pulse settings, using defaults");
495 }
496
497 /* Initialize GPIO for pulse input */
498 err = energy_pulse_gpio_init();
499 if (err) {
500 LOG_ERR("Failed to initialize GPIO for pulse input, error: %d", err);
501 }
502
503 /* Attach ZBUS subscriber to thread */
504 zbus_obs_attach_to_thread(&energy_pulse_subscriber);
505
506 /* Main loop */
507 while (true) {
508 err = task_wdt_feed(task_wdt_id);
509 if (err) {
510 LOG_ERR("Failed to feed task watchdog, error: %d", err);
511 return;
512 }
513
514 /* Wait for message (MSG_SUBSCRIBER gets message copy directly) */
515 err = zbus_sub_wait_msg(&energy_pulse_subscriber, &chan, msg_buf,
516 zbus_wait_ms);
517 if (err == -ENOMSG) {
518 continue;
519 } else if (err) {
520 LOG_ERR("zbus_sub_wait_msg, error: %d", err);
521 return;
522 }
523
524 /* Process message based on channel */
525 if (chan == &DATA_PUBLISH_REQUEST_CHAN) {
526 const struct timing_event_msg *publish_req =
527 MSG_TO_TIMING_EVENT_MSG_PTR(msg_buf);
528 LOG_DBG("Received publish request with timestamp: %lld ms",
529 publish_req->uptime_ms);
530
531 /* Publish current energy pulse data */
532 energy_pulse_msg_publish();
533
534 /* Save energy counters to flash for persistence */
535 err = save_energy_counters();
536 if (err) {
537 LOG_WRN("Failed to persist energy counters, error: %d", err);
538 }
539 } else if (chan == &REBOOT_CHAN) {
540 const struct reboot_msg *reboot = MSG_TO_REBOOT_MSG_PTR(msg_buf);
541 LOG_DBG("Received reboot message with type: %d", reboot->type);
542
543 /* Save energy counters before shutdown/reboot */
544 if (reboot->type == SHUTDOWN_REQ) {
545 LOG_INF("Shutdown requested, saving energy counters...");
546 err = save_energy_counters();
547 if (err) {
548 LOG_ERR("Failed to save energy counters during shutdown, error: %d",
549 err);
550 }
551
552 /* Send shutdown acknowledgement */
553 shutdown_acknowledge_publish(&energy_pulse_subscriber);
554 }
555 }
556 }
557}
