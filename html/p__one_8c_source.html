258/* P1 Rolling Statistics
259 * Updated on each DATA_ACQUISITION_TRIGGER_CHAN event (e.g., every 30s)
260 * Published and reset on DATA_PUBLISH_REQUEST_CHAN event (e.g., every 10min)

262 * Separate tracking for consumed (from grid) and delivered (to grid) power.

308static int32_t obis_fixed_point_to_milli(obis_fixed_point_t fixed_point)
309{
310 /* Multiply whole part by 1000 to get milli-units */
311 int32_t milli = fixed_point.whole * 1000;
312
313 /* Add fractional part: frac is stored as 1e-6 (6 decimals)
314 * Divide by 1000 to convert to milli (keeps 3 decimal places)
315 * Example: frac=456789 456789/1000 = 456 milli-units */
316 milli += fixed_point.frac / 1000;
317
318 return milli;
319}

353static void update_rolling_stats(void)
354{
355 /* Get instantaneous power in mW (from OBIS) - pure int32_t, no FPU! */
356 int32_t power_consumed_w =
357 obis_fixed_point_to_milli(p1_data.power_delivered_kw);
358 int32_t power_produced_w =
359 obis_fixed_point_to_milli(p1_data.power_received_kw);
360
361 int64_t current_timestamp = (int64_t)p1_data.timestamp_epoch_localtime;
362
363 p1_stats.sample_count++;
364
365 /* Update CONSUMED statistics (from grid, usage patterns) */
366 p1_stats.consumed_sum_w += power_consumed_w;
367
368 if (p1_stats.sample_count == 1 ||
369 power_consumed_w > p1_stats.consumed_max_w) {
370 p1_stats.consumed_max_w = power_consumed_w;
371 p1_stats.consumed_max_timestamp = current_timestamp;
372 }
373
374 if (p1_stats.sample_count == 1 ||
375 power_consumed_w < p1_stats.consumed_min_w) {
376 p1_stats.consumed_min_w = power_consumed_w;
377 }
378
379 /* Update DELIVERED statistics (to grid, production patterns) */
380 p1_stats.delivered_sum_w += power_produced_w;
381
382 if (p1_stats.sample_count == 1 ||
383 power_produced_w > p1_stats.delivered_max_w) {
384 p1_stats.delivered_max_w = power_produced_w;
385 p1_stats.delivered_max_timestamp = current_timestamp;
386 }
387
388 if (p1_stats.sample_count == 1 ||
389 power_produced_w < p1_stats.delivered_min_w) {
390 p1_stats.delivered_min_w = power_produced_w;
391 }
392
393 LOG_DBG("Stats: consumed=%d W, delivered=%d W, samples=%u, "
394 "avg: cons=%lld/del=%lld W, max: cons=%d/del=%d W",
395 power_consumed_w, power_produced_w, p1_stats.sample_count,
396 p1_stats.consumed_sum_w / p1_stats.sample_count,
397 p1_stats.delivered_sum_w / p1_stats.sample_count,
398 p1_stats.consumed_max_w, p1_stats.delivered_max_w);
399}

407static void publish_p1_data(void)
408{
409 struct p_one_msg msg = { 0 };
410
411 /* Fill cumulative energy values - convert kWh to mWh (int32_t, no FPU!) */
412 msg.timestamp = (int64_t)p1_data.timestamp_epoch_localtime;
413 msg.energy_consumed_wh_tariff1 =
414 obis_fixed_point_to_milli(p1_data.kwh1_consumed);
415 msg.energy_consumed_wh_tariff2 =
416 obis_fixed_point_to_milli(p1_data.kwh2_consumed);
417 msg.energy_delivered_wh_tariff1 =
418 obis_fixed_point_to_milli(p1_data.kwh1_produced);
419 msg.energy_delivered_wh_tariff2 =
420 obis_fixed_point_to_milli(p1_data.kwh2_produced);
421
422 /* Fill power statistics - CONSUMED (if we have samples) */
423 if (p1_stats.sample_count > 0) {
424 msg.power_consumed_avg_w =
425 (int32_t)(p1_stats.consumed_sum_w / p1_stats.sample_count);
426 msg.power_consumed_max_w = p1_stats.consumed_max_w;
427 msg.power_consumed_max_timestamp = p1_stats.consumed_max_timestamp;
428 msg.power_consumed_min_w = p1_stats.consumed_min_w;
429
430 /* Fill power statistics - DELIVERED */
431 msg.power_delivered_avg_w =
432 (int32_t)(p1_stats.delivered_sum_w / p1_stats.sample_count);
433 msg.power_delivered_max_w = p1_stats.delivered_max_w;
434 msg.power_delivered_max_timestamp = p1_stats.delivered_max_timestamp;
435 msg.power_delivered_min_w = p1_stats.delivered_min_w;
436 } else {
437 /* No samples - zero everything */
438 msg.power_consumed_avg_w = 0;
439 msg.power_consumed_max_w = 0;
440 msg.power_consumed_max_timestamp = 0;
441 msg.power_consumed_min_w = 0;
442
443 msg.power_delivered_avg_w = 0;
444 msg.power_delivered_max_w = 0;
445 msg.power_delivered_max_timestamp = 0;
446 msg.power_delivered_min_w = 0;
447 }
448
449 /* Publish to ZBUS */
450 int err = zbus_chan_pub(&P_ONE_CHAN, &msg, K_SECONDS(1));
451 if (err) {
452 LOG_ERR("Failed to publish P1 data: %d", err);
453 return;
454 }
455
456 LOG_INF("Published P1: consumed=%d+%d Wh, delivered=%d+%d Wh, "
457 "power avg: cons=%d/del=%d W (samples=%u)",
458 msg.energy_consumed_wh_tariff1, msg.energy_consumed_wh_tariff2,
459 msg.energy_delivered_wh_tariff1, msg.energy_delivered_wh_tariff2,
460 msg.power_consumed_avg_w, msg.power_delivered_avg_w,
461 p1_stats.sample_count);
462
463 /* Reset statistics for next period */
464 reset_rolling_stats();
465}

483/* Parse OBIS fixed-point value from buffer */
484static int raw_parse_fixed_point(const uint8_t *buf, obis_value_format_t format,
485 obis_fixed_point_t *result)
486{
487 /* Initialize result to zero */
488 result->whole = 0;
489 result->frac = 0;
490
491 /* Use sscanf to parse the fixed-point number based on the format */
492 switch (format) {
493 case F3_0_0:
494 sscanf(buf, "(%3d", &result->whole);
495 result->frac = 0;
496 break;
497 case F5_3_3:
498 sscanf(buf, "(%2d.%3d", &result->whole, &result->frac);
499 break;
500 case F8_2_2:
501 sscanf(buf, "(%6d.%2d", &result->whole, &result->frac);
502 result->frac *= 10;
503 break;
504 case F8_3_3:
505 sscanf(buf, "(%5d.%3d", &result->whole, &result->frac);
506 break;
507 case F9_3_3:
508 sscanf(buf, "(%6d.%3d", &result->whole, &result->frac);
509 break;
510 case Fn_3_3:
511 LOG_DBG("Fn_3_3 format not implemented yet");
512 return -EINVAL;
513 default:
514 LOG_ERR("Unrecognised format in raw_parse_fixed_point");
515 return -EINVAL;
516 }
517
518 /* Scale fractional part to 1e-6 */
519 result->frac *= 1000;
520
521 return 0;
522}

538 LOG_ERR("Result buffer too small for S4 format");
539 return -EINVAL;
540 }

542 break;

544 if (result_len < 97) {
545 LOG_ERR("Result buffer too small for S96 format");
546 return -EINVAL;
547 }

549 break;
550 default:
551 LOG_ERR("Unrecognised format in raw_parse_string");
552 return -EINVAL;
553 }
554
555 return 0;

558/* Parse OBIS timestamp value from buffer */
559static int raw_parse_timestamp(const uint8_t *buf, obis_value_format_t format,
560 uint64_t *result)
561{
562 /* Initialize result to zero */
563 *result = 0;
564
565 /* Use sscanf to parse the timestamp based on the format */
566 switch (format) {
567 case TST:
568 uint32_t year, month, day, hour, minute, second;
569 char dst;
570
571 /* LOG_DBG("Raw timestamp: %.*s", OBIS_TIMESTAMP_LENGTH, buf + 1); */
572 sscanf(buf, "(%2d%2d%2d%2d%2d%2d%c", &year, &month, &day, &hour,
573 &minute, &second, &dst);
574 year += 2000;
575
576 struct tm p1_localtime = {
577 .tm_year = year - TIME_UTILS_BASE_YEAR,
578 .tm_mon = month - 1,
579 .tm_mday = day,
580 .tm_hour = hour,
581 .tm_min = minute,
582 .tm_sec = second,
583 .tm_isdst = 0, /* ignored by timeutil_timegm64 */
584 };
585
586 uint64_t epoch = timeutil_timegm64(&p1_localtime);
587
588 /* Account for DST ourselves */
589 if (dst == 'S' && epoch > 3600) {
590 epoch -= 3600;
591 }
592
593 *result = epoch;
594 break;
595 default:
596 LOG_ERR("Unrecognised format in raw_parse_timestamp");
597 return -EINVAL;
598 }
599
600 return 0;
601}

605 size_t max_len, obis_fixed_point_t *result)

759/* Calculate CRC16 checksum */
760static uint16_t crc16_calculate(const uint8_t *buf, size_t len)
761{
762 uint16_t crc = 0x0000;
763 for (size_t i = 0; i < len; i++) {
764 crc = (crc >> 8) ^ crc16_lut[(crc ^ buf[i]) & 0xFF];
765 }
766 return crc;
767}

769/* Parse CRC from DSMR telegram with bounds checking */
770static int parse_crc_safe(const uint8_t *buf, size_t len, uint16_t *crc_out)
771{
772 /* Bounds checking - ensure buffer is long enough */
773 if (len < DSMR_STR_CRC_OFFSET_FROM_END) {
774 LOG_ERR("Buffer too short for CRC parsing: len=%zu, need=%d", len,
775 DSMR_STR_CRC_OFFSET_FROM_END);
776 return -EINVAL;
777 }
778
779 /* Find CRC start position */
780 const uint8_t *crc_start = buf + len - DSMR_STR_CRC_OFFSET_FROM_END;
781
782 /* Copy exactly 4 hex characters to controlled buffer */
783 char crc_str[5]; /* 4 chars + null terminator */
784 memcpy(crc_str, crc_start, DSMR_STR_CRC_LEN);
785 crc_str[4] = '\0';
786
787 /* Validate that we have valid hex characters */
788 for (int i = 0; i < DSMR_STR_CRC_LEN; i++) {
789 char c = crc_str[i];
790 if (!((c >= '0' && c <= '9') || (c >= 'A' && c <= 'F') ||
791 (c >= 'a' && c <= 'f'))) {
792 LOG_ERR("Invalid hex character in CRC at pos %d: 0x%02X ('%c')", i,
793 (uint8_t)c, (c >= 32 && c <= 126) ? c : '?');
794 return -EINVAL;
795 }
796 }
797
798 /* Parse CRC using sscanf on controlled buffer */
799 uint16_t crc_val;
800 int parsed = sscanf(crc_str, "%4hX", &crc_val);
801
802 if (parsed != 1) {
803 LOG_ERR("Failed to parse CRC from: '%.4s'", crc_str);
804 return -EINVAL;
805 }
806
807 *crc_out = crc_val;
808 LOG_DBG("Parsed CRC: %04X from '%.4s'", crc_val, crc_str);
809 return 0;
810}

820static int p_one_acquire_data(void)
821{
822 LOG_INF("Acquiring data from P1 port");
823 int err;
824
825 /* Reset RX buffer counter */
826 uart_user_data.total_bytes_received = 0;
827
828 /* Reset semaphore to ensure clean state (discard any leftover signals) */
829 k_sem_reset(&p_one_uart_rx_sem);
830
831 /* Enable the UART - timeout event is triggered when the pause between
832 * received bytes exceeds half the maximum telegram length's time */
833 LOG_DBG("Enabling RX, address %p", (void *)p_one_uart_rx_buf);
834 err = uart_rx_enable(uart_dev, p_one_uart_rx_buf, UART_RX_BUF_SIZE,
835 P_ONE_UART_MAX_TELEGRAM_TIMEOUT_USEC * 0.5);
836 if (err) {
837 LOG_ERR("Failed to enable RX, error: %d", err);
838 return -EIO;
839 }
840
841 /* Assert Data Request */
842 gpio_pin_set_dt(&p_one_data_request, 1);
843 LOG_DBG("Data_Request pin asserted");
844
845 /* Wait for UART to be disabled */
846 err = k_sem_take(&p_one_uart_rx_sem,
847 K_MSEC(P_ONE_DATA_REQUEST_RESPONSE_TIMEOUT_MSEC));
848
849 /* Deassert Data Request */
850 gpio_pin_set_dt(&p_one_data_request, 0);
851 LOG_DBG("Data_Request pin deasserted");
852
853 if (err != 0) {
854 /* Semaphore timeout: P1 device did not respond within 10+ seconds */
855 LOG_WRN("UART timed out - no response from P1 device");
856
857 /* Disable UART to clean up */
858 err = uart_rx_disable(uart_dev);
859 if (err) {
860 LOG_ERR("Failed to disable RX after timeout, error: %d", err);
861 return -EIO;
862 }
863
864 /* Wait for UART_RX_DISABLED event to complete cleanup and avoid race
865 * condition with next cycle */
866 err = k_sem_take(&p_one_uart_rx_sem, K_MSEC(100));
867 if (err) {
868 LOG_ERR("UART_RX_DISABLED event did not arrive after "
869 "manual disable");
870 }
871
872 /* No valid data received */
873 return -ETIMEDOUT;
874 }
875
876 /* UART_RX_DISABLED event received within timeout */
877 /* Check if we have valid data (happy path) or error occurred */
878 if (uart_user_data.total_bytes_received == 0) {
879 LOG_WRN("UART stopped due to error, no valid data received");
880 return -EIO;
881 }
882
883 /* Happy path: valid data received! */
884 LOG_DBG("UART RX completed successfully, total bytes received: %d",
885 uart_user_data.total_bytes_received);
886
887 return 0;
888}

896static void p_one_parse_data(void)
897{
898 LOG_INF("Parsing P_ONE RX data");
899
900 size_t len = uart_user_data.total_bytes_received;
901 uint8_t *buf = p_one_uart_rx_buf;
902
903 /* Sanity check, still too short actually */
904 if (len < DSMR_MIN_LEN) {
905 LOG_WRN("DSMR data too short, length: %zu", len);
906 return;
907 }
908
909 /* Check header and footer (/), (!) */
910 if ((buf[DSMR_CHAR_HEADER_OFFSET] != DSMR_CHAR_HEADER) ||
911 (buf[len - DSMR_CHAR_FOOTER_OFFSET_FROM_END] != DSMR_CHAR_FOOTER) ||
912 (DSMR_CHAR_FOOTER_OFFSET_FROM_END > len)) {
913 LOG_WRN("DSMR header and footer invalid");
914 LOG_DBG("Header: %c (should be %c), Footer: %c (should be %c), "
915 "Length: %zu",
916 buf[DSMR_CHAR_HEADER_OFFSET], DSMR_CHAR_HEADER,
917 buf[len - DSMR_CHAR_FOOTER_OFFSET_FROM_END], DSMR_CHAR_FOOTER,
918 len);
919 return;
920 }
921
922 /* Check CRC-16 using safe parsing */
923 uint16_t crc_calculated = crc16_calculate(
924 &buf[DSMR_CHAR_HEADER_OFFSET], len - DSMR_STR_CRC_OFFSET_FROM_END);
925
926 uint16_t crc_received;
927 int crc_parse_result = parse_crc_safe(buf, len, &crc_received);
928 if (crc_parse_result != 0) {
929 LOG_WRN("Failed to parse CRC from telegram, error: %d",
930 crc_parse_result);
931 return;
932 }
933
934 if (crc_calculated != crc_received) {
935 LOG_WRN("DSMR CRC mismatch, calculated: %04X, received: %04X",
936 crc_calculated, crc_received);
937 return;
938 }
939 LOG_DBG("DSMR header, footer and CRC valid");
940
941 int i = DSMR_CHAR_HEADER_OFFSET + 1;
942
943 /* Find start of data (CRLFCRLF) */
944 for (; i < len - DSMR_STR_START_OF_DATA_LEN; i++) {
945 if (strncmp(&buf[i], DSMR_STR_START_OF_DATA,
946 DSMR_STR_START_OF_DATA_LEN) == 0) {
947 /* Skip the header */
948 i = i + DSMR_STR_START_OF_DATA_LEN;
949 LOG_DBG("Found start of data at [%d]", i);
950 break;
951 } else if (i >= len - DSMR_MIN_LEN) {
952 LOG_WRN("DSMR Header invalid");
953 return;
954 }
955 }
956
957 /* Check the P1 Version */
958 if (OBIS_FIND(&buf[i], DSMR_OBIS_P1_VERSION)) {
959 obis_parse_string(
960 &buf[i], DSMR_OBIS_P1_VERSION,
961 (obis_lut[DSMR_OBIS_P1_VERSION].len + 2 + 2 +
962 DSMR_STR_LINE_END_LEN), /* Sorry, not dynamic to S2 length */
963 p1_extra_data.version, ARRAY_SIZE(p1_extra_data.version));
964
965 if (p1_extra_data.version[0] == '5' &&
966 p1_extra_data.version[1] == '0') {
967 LOG_DBG("P1 version is 5.0 (%2s)", p1_extra_data.version);
968 } else if (p1_extra_data.version[0] == '4') {
969 LOG_DBG("P1 version is 4.X (%2s)", p1_extra_data.version);
970 } else {
971 LOG_WRN("P1 version is not 4.X or 5.0, literally \"%2s\"",
972 p1_extra_data.version);
973 return;
974 }
975 } else {
976 LOG_WRN("P1 version not found");
977 return;
978 }
979
980 /* Iterate as long as we don't reach the footer */
981 for (; i < len - DSMR_CHAR_FOOTER_OFFSET_FROM_END;) {
982 int line_len = 0; /* Length of the line, including CRLF */
983
984 /* Iterate until we find the position of the line ending (CRLF)
985 */
986 for (int j = i; (j < len - DSMR_STR_LINE_END_LEN) &&
987 (j < len - DSMR_CHAR_FOOTER_OFFSET_FROM_END);
988 j++) {
989 /* Check for CRLF */
990 if (strncmp(&buf[j], DSMR_STR_LINE_END, DSMR_STR_LINE_END_LEN) ==
991 0) {
992 /* Good, the position is at [j] */
993
994 /* Calculate line length including CRLF */
995 line_len = j - i + DSMR_STR_LINE_END_LEN;
996
997 /* This means the line we found is only CRLF */
998 if (line_len == DSMR_STR_LINE_END_LEN) {
999 LOG_WRN("Found empty line at [%d], length %d", i, line_len);
1000
1001 /* Debug output of surrounding characters */
1002 if (i >= 4) {
1003 LOG_DBG("Line contents (surrounding): "
1004 "'%c'_'%c'_'%c'_'%c'_'%c'_'%c'_'%c'_'%c'",
1005 buf[i - 4], buf[i - 3], buf[i - 2], buf[i - 1],
1006 buf[i], buf[i + 1], buf[i + 2], buf[i + 3]);
1007 }
1008 } else {
1009 /* LOG_DBG("Found line at [%d], length %d", i,
1010 * line_len); */
1011 }
1012
1013 break; /* Exit inner loop, go to processing */
1014 }
1015 }
1016
1017 /* Handle cases where no valid data line was found:
1018 *
1019 * Scenario 1: line_len == 0
1020 * - Inner loop found no CRLF within allowed boundaries
1021 * - Occurs when: corrupted data, incomplete telegram, or at end of
1022 * valid data region
1023 * - line_len stays at initial value 0 since no CRLF match was found
1024 * - Advance by 1 byte to prevent infinite loop and try next position
1025 *
1026 * Scenario 2: line_len == 2 (== DSMR_STR_LINE_END_LEN)
1027 * - Found empty line containing only CRLF ("\r\n")
1028 * - This is valid DSMR format but contains no parseable data
1029 * - Advance by 2 bytes to skip over the empty line
1030 *
1031 * Both scenarios: Skip processing and continue to next iteration
1032 */
1033 if (line_len <= DSMR_STR_LINE_END_LEN) {
1034 i += (line_len > 0) ? line_len : 1;
1035 continue;
1036 }
1037
1038 /* Copy line to temporary buffer for safe string operations */
1039 /* Calculate actual data length (excluding CRLF) */
