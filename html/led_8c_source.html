218static void apply_led_pattern(const struct led_pattern *pattern)
219{
220 int err;
221
222 /* Cancel current blink */
223 k_work_cancel_delayable(&work_ledblink);
224
225 /* Start with LEDs on */
226 led_on_state = true;
227 set_led_color(pattern->red, pattern->green, pattern->blue);
228
229 /* Schedule blink if period is non-zero */
230 if (pattern->period > 0) {
231 err = k_work_schedule(&work_ledblink, K_MSEC(pattern->period / 2));
232 if (err < 0) {
233 LOG_ERR("Failed to schedule LED blink: %d", err);
234 }
235 }
236}

244static enum led_status_priority
245determine_led_status(const struct led_module_state *state)
246{
247 /* Priority order (highest first) */
248
249 /* FOTA has highest priority */
250 if (state->fota_active) {
251 return LED_STATUS_FOTA;
252 }
253
254 if (state->fota_error) {
255 return LED_STATUS_FOTA_ERROR;
256 }
257
258 /* Wait for modem initialization */
259 if (!state->modem_initialized) {
260 return LED_STATUS_WAIT;
261 }
262
263 if (!state->network_connected) {
264 return LED_STATUS_NETWORK_ERROR;
265 }
266
267 if (!state->cloud_ready) {
268 return LED_STATUS_CLOUD_ERROR;
269 }
270
271 if (state->acquiring) {
272 return LED_STATUS_ACQUIRING;
273 }
274
275 // if (state->cloud_tx) {
276 // return LED_STATUS_CLOUD_TX;
277 // }
278
279 /* Default to IDLE if everything is OK */
280 return LED_STATUS_IDLE;
281}

290static void update_led_pattern(struct led_module_state *state)
291{
292 enum led_status_priority new_status = determine_led_status(state);
293
294 /* Apply pattern only if status changed */
295 if (new_status != current_status) {
296 LOG_DBG("LED status changed to %d", new_status);
297 current_status = new_status;
298 apply_led_pattern(&led_patterns[new_status]);
299 }
300}

315static void ledblink_handler(struct k_work *work)
316{
317 ARG_UNUSED(work);
318
319 const struct led_pattern *pattern = &led_patterns[current_status];
320 int err;
321
322 /* Toggle LED state */
323 led_on_state = !led_on_state;
324
325 if (led_on_state) {
326 set_led_color(pattern->red, pattern->green, pattern->blue);
327 } else {
328 set_led_color(0, 0, 0);
329 }
330
331 /* Schedule next toggle */
332 err = k_work_schedule(&work_ledblink, K_MSEC(pattern->period / 2));
333 if (err < 0) {
334 LOG_ERR("Failed to schedule LED blink: %d", err);
335 }
336}

347{
348 struct k_work_delayable *dwork = k_work_delayable_from_work(work);

370static void process_network_msg(struct led_module_state *state,
371 const struct network_msg *msg)
372{

374 state->modem_initialized = true;

376
377 if (network_is_connected(msg)) {
378 state->network_connected = true;

380 state->network_connected = false;
381 /* Cancel acquisition if network drops */
382 if (state->acquiring) {
383 state->acquiring = false;
384 k_work_cancel_delayable(&state->acquiring_timeout);
385 }
386 }

438static void process_fota_msg(struct led_module_state *state,
439 const struct fota_msg *fota)
440{
441 LOG_DBG("FOTA status: %d", fota->status);
442
443 /* Set fota_active and fota_error flags based on status */
444 switch (fota->status) {
445 case FOTA_STATUS_DOWNLOADING:
446 state->fota_active = true;
447 state->fota_error = false;
448 LOG_INF("FOTA download started - LED: 4Hz green blink");
449 break;
450 case FOTA_STATUS_READY:
451 state->fota_active = false;
452 state->fota_error = false;
453 LOG_INF("FOTA download complete - returning to normal LED pattern");
454 break;
455 case FOTA_STATUS_ERROR:
456 state->fota_active = false;
457 state->fota_error = true;
458 LOG_ERR("FOTA error - LED: constant red");
459 break;
460 case FOTA_STATUS_IDLE:
461 state->fota_active = false;
462 state->fota_error = false;
463 LOG_DBG("FOTA idle - returning to normal LED pattern");
464 break;
465 default:
466 LOG_WRN("Unknown FOTA status: %d", fota->status);
467 break;
468 }
469}

482static void led_thread(void)
483{
484 struct led_module_state state = {
485 .modem_initialized = false,
486 .network_connected = false,
487 .cloud_ready = false,
488 .acquiring = false,
489 .fota_active = false,
490 .fota_error = false,
491 };
492 const struct zbus_channel *chan;
493 uint8_t msg_buf[MAX_MSG_SIZE];
494 int err;
495 int task_wdt_id;
496 const uint32_t wdt_timeout_ms =
497 CONFIG_LED_MODULE_WATCHDOG_TIMEOUT_SECONDS * MSEC_PER_SEC;
498 const uint32_t msg_processing_ms =
499 CONFIG_LED_MODULE_MSG_PROCESSING_TIME_SECONDS * MSEC_PER_SEC;
500 const k_timeout_t zbus_wait_ms = K_MSEC(wdt_timeout_ms - msg_processing_ms);
501
502 LOG_DBG("LED thread started - WDT: %d s, Processing: %d s, ZBUS Wait: %d s",
503 wdt_timeout_ms / MSEC_PER_SEC, msg_processing_ms / MSEC_PER_SEC,
504 (wdt_timeout_ms - msg_processing_ms) / MSEC_PER_SEC);
505
506 task_wdt_id = task_wdt_util_setup(wdt_timeout_ms, task_wdt_callback,
507 (void *)k_current_get());
508 if (task_wdt_id < 0) {
509 LOG_ERR("Failed to setup watchdog");
510 return;
511 }
512
513 /* Attach ZBUS subscriber to thread for priority handling */
514 zbus_obs_attach_to_thread(&led_subscriber);
515
516 /* Initialize acquiring timeout work item */
517 k_work_init_delayable(&state.acquiring_timeout, acquiring_timeout_handler);
518
519 /* Apply initial pattern (already set in current_status initialization) */
520 apply_led_pattern(&led_patterns[current_status]);
521
522 while (true) {
523 err = task_wdt_feed(task_wdt_id);
524 if (err) {
525 LOG_ERR("Failed to feed task watchdog, error: %d", err);
526 return;
527 }
528
529 /* Wait for message (MSG_SUBSCRIBER gets message copy directly) */
530 err = zbus_sub_wait_msg(&led_subscriber, &chan, msg_buf, zbus_wait_ms);
531 if (err == -ENOMSG) {
532 continue;
533 } else if (err) {
534 LOG_ERR("zbus_sub_wait_msg failed: %d", err);
535 return;
536 }
537
538 /* Process message based on channel */
539 if (chan == &NETWORK_CHAN) {
540 const struct network_msg *msg = MSG_TO_NETWORK_MSG_PTR(msg_buf);
541 process_network_msg(&state, msg);
542 } else if (chan == &CLOUD_CHAN) {
543 const struct cloud_msg *msg = MSG_TO_CLOUD_MSG_PTR(msg_buf);
544 process_cloud_msg(&state, msg);
545 } else if (chan == &FOTA_CHAN) {
546 const struct fota_msg *fota = MSG_TO_FOTA_MSG_PTR(msg_buf);
547 process_fota_msg(&state, fota);
548 } else if (chan == &TRIGGER_CHAN) {
549 const struct trigger_msg *trigger = MSG_TO_TRIGGER_MSG_PTR(msg_buf);
550 process_trigger_msg(&state, trigger);
551 } else {
552 LOG_WRN("Received message from unknown channel");
553 }
554
555 /* Update LED pattern based on new state */
556 update_led_pattern(&state);
557 }
558}
