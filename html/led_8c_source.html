272static void apply_led_pattern(const struct led_pattern *pattern)
273{
274 int err;
275
276 /* Cancel current blink */
277 k_work_cancel_delayable(&work_ledblink);
278
279 /* Check if pattern is constant on (period_off = 0) */
280 if (pattern->period_off == 0) {
281 led_on_state = true;
282 set_led_color(pattern->red, pattern->green, pattern->blue);
283 return;
284 }
285
286 /* Check if pattern is constant off (period_on = 0) */
287 if (pattern->period_on == 0) {
288 led_on_state = false;
289 set_led_color(0, 0, 0);
290 return;
291 }
292
293 /* Start with LEDs on */
294 led_on_state = true;
295 set_led_color(pattern->red, pattern->green, pattern->blue);
296
297 /* Schedule blink with period_on duration */
298 err = k_work_schedule(&work_ledblink, K_MSEC(pattern->period_on));
299 if (err < 0) {
300 LOG_ERR("Failed to schedule LED blink: %d", err);
301 }
302}

310static enum led_status_priority
311determine_led_status(const struct led_module_state *state)
312{
313 /* Priority order (highest first) */
314
315 /* FOTA has highest priority */
316 if (state->fota_active) {
317 return LED_STATUS_FOTA;
318 }
319
320 if (state->fota_error) {
321 return LED_STATUS_FOTA_ERROR;
322 }
323
324 /* Wait for modem initialization */
325 if (!state->modem_initialized) {
326 return LED_STATUS_WAIT;
327 }
328
329 /* Network state machine */
330 switch (state->net_state) {
331 case NETWORK_STATE_NEVER_CONNECTED:
332 case NETWORK_STATE_CONNECTING:
333 return LED_STATUS_CONNECTING;
334
335 case NETWORK_STATE_ERROR:
336 return LED_STATUS_NETWORK_ERROR;
337
338 case NETWORK_STATE_CONNECTED:
339 /* Fall through to check cloud/other states */
340 break;
341 }
342
343 if (!state->cloud_ready) {
344 return LED_STATUS_CLOUD_ERROR;
345 }
346
347 if (state->acquiring) {
348 return LED_STATUS_ACQUIRING;
349 }
350
351 // if (state->cloud_tx) {
352 // return LED_STATUS_CLOUD_TX;
353 // }
354
355 /* Default to IDLE if everything is OK */
356 return LED_STATUS_IDLE;
357}

366static void update_led_pattern(struct led_module_state *state)
367{
368 enum led_status_priority new_status = determine_led_status(state);
369
370 /* If we're in minimum display time, just remember the new status */
371 if (min_display_active) {
372 if (new_status != pending_status) {
373 LOG_DBG("LED status change pending: %s (%d) - currently showing %s (%d)",
374 led_status_names[new_status], new_status,
375 led_status_names[current_status], current_status);
376 pending_status = new_status;
377 }
378 return;
379 }
380
381 /* Apply pattern only if status changed */
382 if (new_status != current_status) {
383 const struct led_pattern *pattern = &led_patterns[new_status];
384
385 LOG_DBG("LED status changed to %s (%d)", led_status_names[new_status],
386 new_status);
387 current_status = new_status;
388 pending_status = new_status;
389 apply_led_pattern(pattern);
390
391 /* Start minimum display timer if configured */
392 if (pattern->min_display_time_ms > 0) {
393 min_display_active = true;
394 k_work_schedule(&work_min_display_timeout,
395 K_MSEC(pattern->min_display_time_ms));
396 }
397 }
398}

413static void ledblink_handler(struct k_work *work)
414{
415 ARG_UNUSED(work);
416
417 const struct led_pattern *pattern = &led_patterns[current_status];
418 int err;
419 uint16_t next_period;
420
421 /* Toggle LED state */
422 led_on_state = !led_on_state;
423
424 if (led_on_state) {
425 set_led_color(pattern->red, pattern->green, pattern->blue);
426 next_period = pattern->period_on;
427 } else {
428 set_led_color(0, 0, 0);
429 next_period = pattern->period_off;
430 }
431
432 /* Schedule next toggle */
433 err = k_work_schedule(&work_ledblink, K_MSEC(next_period));
434 if (err < 0) {
435 LOG_ERR("Failed to schedule LED blink: %d", err);
436 }
437}

512static void process_network_msg(struct led_module_state *state,
513 const struct network_msg *msg)
514{
515 if (network_is_modem_initialized(msg)) {
516 state->modem_initialized = true;
517 /* Transition: NEVER_CONNECTED CONNECTING */
518 if (state->net_state == NETWORK_STATE_NEVER_CONNECTED) {
519 LOG_INF("Modem initialized, entering CONNECTING state");
520 state->net_state = NETWORK_STATE_CONNECTING;
521 state->disconnect_timestamp = k_uptime_get();
522 }
523 }
524
525 if (network_is_connected(msg)) {
526 state->network_connected = true;
527 /* Transition: Any state CONNECTED */
528 LOG_INF("Network connected (state was: %d)", state->net_state);
529 state->net_state = NETWORK_STATE_CONNECTED;
530 state->disconnect_timestamp = 0;
531
532 } else if (network_is_disconnected(msg)) {
533 state->network_connected = false;
534
535 switch (state->net_state) {
536 case NETWORK_STATE_NEVER_CONNECTED:
537 case NETWORK_STATE_CONNECTING: {
538 /* Check timeout: 120s since entering CONNECTING state */
539 int64_t elapsed_ms = k_uptime_get() - state->disconnect_timestamp;
540 if (elapsed_ms > K_SECONDS(120).ticks) {
541 LOG_WRN("Network connect timeout (%lld ms), entering ERROR state",
542 elapsed_ms);
543 state->net_state = NETWORK_STATE_ERROR;
544 } else {
545 LOG_DBG("Still connecting (%lld ms / 120s)", elapsed_ms / 1000);
546 }
547 break;
548 }
549
550 case NETWORK_STATE_CONNECTED:
551 /* Lost connection - start reconnecting */
552 LOG_INF("Network disconnected, entering CONNECTING state (reconnect)");
553 state->net_state = NETWORK_STATE_CONNECTING;
554 state->disconnect_timestamp = k_uptime_get();
555 break;
556
557 case NETWORK_STATE_ERROR:
558 /* Already in error state, stay there */
559 break;
560
561 default:
562 break;
563 }
564
565 /* Cancel acquisition if network drops */
566 if (state->acquiring) {
567 state->acquiring = false;
568 k_work_cancel_delayable(&state->acquiring_timeout);
569 }
570 }
571}

622static void process_fota_msg(struct led_module_state *state,
623 const struct fota_msg *fota)
624{
625 LOG_DBG("FOTA status: %d", fota->status);
626
627 /* Set fota_active and fota_error flags based on status */
628 switch (fota->status) {
629 case FOTA_STATUS_DOWNLOADING:
630 state->fota_active = true;
631 state->fota_error = false;
632 LOG_INF("FOTA download started - LED: 4Hz green blink");
633 break;
634 case FOTA_STATUS_READY:
635 state->fota_active = false;
636 state->fota_error = false;
637 LOG_INF("FOTA download complete - returning to normal LED pattern");
638 break;
639 case FOTA_STATUS_ERROR:
640 state->fota_active = false;
641 state->fota_error = true;
642 LOG_ERR("FOTA error - LED: constant red");
643 break;
644 case FOTA_STATUS_IDLE:
645 state->fota_active = false;
646 state->fota_error = false;
647 LOG_DBG("FOTA idle - returning to normal LED pattern");
648 break;
649 default:
650 LOG_WRN("Unknown FOTA status: %d", fota->status);
651 break;
652 }
653}

666static void led_thread(void)
667{
668 struct led_module_state state = {
669 .modem_initialized = false,
670 .network_connected = false,
671 .cloud_ready = false,
672 .acquiring = false,
673 .fota_active = false,
674 .fota_error = false,
675 .net_state = NETWORK_STATE_NEVER_CONNECTED,
676 .disconnect_timestamp = 0,
677 };
678 const struct zbus_channel *chan;
679 uint8_t msg_buf[MAX_MSG_SIZE];
680 int err;
681 int task_wdt_id;
682 const uint32_t wdt_timeout_ms =
683 CONFIG_LED_MODULE_WATCHDOG_TIMEOUT_SECONDS * MSEC_PER_SEC;
684 const uint32_t msg_processing_ms =
685 CONFIG_LED_MODULE_MSG_PROCESSING_TIME_SECONDS * MSEC_PER_SEC;
686 const k_timeout_t zbus_wait_ms = K_MSEC(wdt_timeout_ms - msg_processing_ms);
687
688 LOG_DBG("LED thread started - WDT: %d s, Processing: %d s, ZBUS Wait: %d s",
689 wdt_timeout_ms / MSEC_PER_SEC, msg_processing_ms / MSEC_PER_SEC,
690 (wdt_timeout_ms - msg_processing_ms) / MSEC_PER_SEC);
691
692 task_wdt_id = task_wdt_util_setup(wdt_timeout_ms, task_wdt_callback,
693 (void *)k_current_get());
694 if (task_wdt_id < 0) {
695 LOG_ERR("Failed to setup watchdog");
696 return;
697 }
698
699 /* Attach ZBUS subscriber to thread for priority handling */
700 zbus_obs_attach_to_thread(&led_subscriber);
701
702 /* Initialize acquiring timeout work item */
703 k_work_init_delayable(&state.acquiring_timeout, acquiring_timeout_handler);
704
705 /* Apply initial pattern (already set in current_status initialization) */
706 apply_led_pattern(&led_patterns[current_status]);
707
708 while (true) {
709 err = task_wdt_feed(task_wdt_id);
710 if (err) {
711 LOG_ERR("Failed to feed task watchdog, error: %d", err);
712 return;
713 }
714
715 /* Wait for message (MSG_SUBSCRIBER gets message copy directly) */
716 err = zbus_sub_wait_msg(&led_subscriber, &chan, msg_buf, zbus_wait_ms);
717 if (err == -ENOMSG) {
718 continue;
719 } else if (err) {
720 LOG_ERR("zbus_sub_wait_msg failed: %d", err);
721 return;
722 }
723
724 /* Process message based on channel */
725 if (chan == &NETWORK_CHAN) {
726 const struct network_msg *msg = MSG_TO_NETWORK_MSG_PTR(msg_buf);
727 process_network_msg(&state, msg);
728 } else if (chan == &CLOUD_CHAN) {
729 const struct cloud_msg *msg = MSG_TO_CLOUD_MSG_PTR(msg_buf);
730 process_cloud_msg(&state, msg);
731 } else if (chan == &FOTA_CHAN) {
732 const struct fota_msg *fota = MSG_TO_FOTA_MSG_PTR(msg_buf);
733 process_fota_msg(&state, fota);
734 } else if (chan == &TRIGGER_CHAN) {
735 const struct trigger_msg *trigger = MSG_TO_TRIGGER_MSG_PTR(msg_buf);
736 process_trigger_msg(&state, trigger);
737 } else {
738 LOG_WRN("Received message from unknown channel");
739 }
740
741 /* Update LED pattern based on new state */
742 update_led_pattern(&state);
743 }
744}
