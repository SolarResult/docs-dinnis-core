179/* Watchdog timeout callback */
180static void task_wdt_callback(int channel_id, void *user_data)
181{
182 static bool reboot_requested = false;
183
184 if (!reboot_requested) {
185 reboot_requested = true;
186 LOG_ERR("Watchdog timeout occurred, Channel ID: %d, Thread: %s",
187 channel_id, k_thread_name_get((k_tid_t)user_data));
188
189 /* Task watchdog timeout is fatal - force immediate reboot */
190 reboot_request_publish(REBOOT_TYPE_ERROR);
191 } else {
192 /* Backup: Force immediate cold reboot if graceful shutdown failed */
193 LOG_ERR("Watchdog timeout re-triggered - forcing immediate reboot");
194 sys_reboot(SYS_REBOOT_COLD);
195 }
196}

278static void apply_led_pattern(const struct led_pattern *pattern)
279{
280 int err;
281
282 /* Cancel current blink */
283 k_work_cancel_delayable(&work_ledblink);
284
285 /* Check if pattern is constant on (period_off = 0) */
286 if (pattern->period_off == 0) {
287 led_on_state = true;
288 set_led_color(pattern->red, pattern->green, pattern->blue);
289 return;
290 }
291
292 /* Check if pattern is constant off (period_on = 0) */
293 if (pattern->period_on == 0) {
294 led_on_state = false;
295 set_led_color(0, 0, 0);
296 return;
297 }
298
299 /* Start with LEDs on */
300 led_on_state = true;
301 set_led_color(pattern->red, pattern->green, pattern->blue);
302
303 /* Schedule blink with period_on duration */
304 err = k_work_schedule(&work_ledblink, K_MSEC(pattern->period_on));
305 if (err < 0) {
306 LOG_ERR("Failed to schedule LED blink: %d", err);
307 }
308}

316static enum led_status_priority
317determine_led_status(const struct led_module_state *state)
318{
319 /* Priority order (highest first) */
320
321 /* FOTA has highest priority */
322 if (state->fota_active) {
323 return LED_STATUS_FOTA;
324 }
325
326 if (state->fota_error) {
327 return LED_STATUS_FOTA_ERROR;
328 }
329
330 /* Wait for modem initialization */
331 if (!state->modem_initialized) {
332 return LED_STATUS_WAIT;
333 }
334
335 /* Network state machine */
336 switch (state->net_state) {
337 case NETWORK_STATE_NEVER_CONNECTED:
338 case NETWORK_STATE_CONNECTING:
339 return LED_STATUS_CONNECTING;
340
341 case NETWORK_STATE_ERROR:
342 return LED_STATUS_NETWORK_ERROR;
343
344 case NETWORK_STATE_CONNECTED:
345 /* Fall through to check cloud/other states */
346 break;
347 }
348
349 if (!state->cloud_ready) {
350 return LED_STATUS_CLOUD_ERROR;
351 }
352
353 if (state->acquiring) {
354 return LED_STATUS_ACQUIRING;
355 }
356
357 // if (state->cloud_tx) {
358 // return LED_STATUS_CLOUD_TX;
359 // }
360
361 /* Default to IDLE if everything is OK */
362 return LED_STATUS_IDLE;
363}

372static void update_led_pattern(struct led_module_state *state)
373{
374 enum led_status_priority new_status = determine_led_status(state);
375
376 /* If we're in minimum display time, just remember the new status */
377 if (min_display_active) {
378 if (new_status != pending_status) {
379 LOG_DBG("LED status change pending: %s (%d) - currently showing %s (%d)",
380 led_status_names[new_status], new_status,
381 led_status_names[current_status], current_status);
382 pending_status = new_status;
383 }
384 return;
385 }
386
387 /* Apply pattern only if status changed */
388 if (new_status != current_status) {
389 const struct led_pattern *pattern = &led_patterns[new_status];
390
391 LOG_DBG("LED status changed to %s (%d)", led_status_names[new_status],
392 new_status);
393 current_status = new_status;
394 pending_status = new_status;
395 apply_led_pattern(pattern);
396
397 /* Start minimum display timer if configured */
398 if (pattern->min_display_time_ms > 0) {
399 min_display_active = true;
400 k_work_schedule(&work_min_display_timeout,
401 K_MSEC(pattern->min_display_time_ms));
402 }
403 }
404}

419static void ledblink_handler(struct k_work *work)
420{
421 ARG_UNUSED(work);
422
423 const struct led_pattern *pattern = &led_patterns[current_status];
424 int err;
425 uint16_t next_period;
426
427 /* Toggle LED state */
428 led_on_state = !led_on_state;
429
430 if (led_on_state) {
431 set_led_color(pattern->red, pattern->green, pattern->blue);
432 next_period = pattern->period_on;
433 } else {
434 set_led_color(0, 0, 0);
435 next_period = pattern->period_off;
436 }
437
438 /* Schedule next toggle */
439 err = k_work_schedule(&work_ledblink, K_MSEC(next_period));
440 if (err < 0) {
441 LOG_ERR("Failed to schedule LED blink: %d", err);
442 }
443}

474{
475 ARG_UNUSED(work);

518static void process_network_msg(struct led_module_state *state,
519 const struct network_msg *msg)
520{
521 if (network_is_modem_initialized(msg)) {
522 state->modem_initialized = true;
523 /* Transition: NEVER_CONNECTED CONNECTING */
524 if (state->net_state == NETWORK_STATE_NEVER_CONNECTED) {
525 LOG_INF("Modem initialized, entering CONNECTING state");
526 state->net_state = NETWORK_STATE_CONNECTING;
527 state->disconnect_timestamp = k_uptime_get();
528 }
529 }
530
531 if (network_is_connected(msg)) {
532 state->network_connected = true;
533 /* Transition: Any state CONNECTED */
534 LOG_INF("Network connected (state was: %d)", state->net_state);
535 state->net_state = NETWORK_STATE_CONNECTED;
536 state->disconnect_timestamp = 0;
537
538 } else if (network_is_disconnected(msg)) {
539 state->network_connected = false;
540
541 switch (state->net_state) {
542 case NETWORK_STATE_NEVER_CONNECTED:
543 case NETWORK_STATE_CONNECTING: {
544 /* Check timeout: 120s since entering CONNECTING state */
545 int64_t elapsed_ms = k_uptime_get() - state->disconnect_timestamp;
546 if (elapsed_ms > K_SECONDS(120).ticks) {
547 LOG_WRN("Network connect timeout (%lld ms), entering ERROR state",
548 elapsed_ms);
549 state->net_state = NETWORK_STATE_ERROR;
550 } else {
551 LOG_DBG("Still connecting (%lld ms / 120s)", elapsed_ms / 1000);
552 }
553 break;
554 }
555
556 case NETWORK_STATE_CONNECTED:
557 /* Lost connection - start reconnecting */
558 LOG_INF("Network disconnected, entering CONNECTING state (reconnect)");
559 state->net_state = NETWORK_STATE_CONNECTING;
560 state->disconnect_timestamp = k_uptime_get();
561 break;
562
563 case NETWORK_STATE_ERROR:
564 /* Already in error state, stay there */
565 break;
566
567 default:
568 break;
569 }
570
571 /* Cancel acquisition if network drops */
572 if (state->acquiring) {
573 state->acquiring = false;
574 k_work_cancel_delayable(&state->acquiring_timeout);
575 }
576 }
577}

628static void process_fota_msg(struct led_module_state *state,
629 const struct fota_msg *fota)
630{
631 LOG_DBG("FOTA status: %d", fota->status);
632
633 /* Set fota_active and fota_error flags based on status */
634 switch (fota->status) {
635 case FOTA_STATUS_DOWNLOADING:
636 state->fota_active = true;
637 state->fota_error = false;
638 LOG_INF("FOTA download started - LED: 4Hz green blink");
639 break;
640 case FOTA_STATUS_READY:
641 state->fota_active = false;
642 state->fota_error = false;
643 LOG_INF("FOTA download complete - returning to normal LED pattern");
644 break;
645 case FOTA_STATUS_ERROR:
646 state->fota_active = false;
647 state->fota_error = true;
648 LOG_ERR("FOTA error - LED: constant red");
649 break;
650 case FOTA_STATUS_IDLE:
651 state->fota_active = false;
652 state->fota_error = false;
653 LOG_DBG("FOTA idle - returning to normal LED pattern");
654 break;
655 default:
656 LOG_WRN("Unknown FOTA status: %d", fota->status);
657 break;
658 }
659}

672static void led_thread(void)
673{
674 struct led_module_state state = {
675 .modem_initialized = false,
676 .network_connected = false,
677 .cloud_ready = false,
678 .acquiring = false,
679 .fota_active = false,
680 .fota_error = false,
681 .net_state = NETWORK_STATE_NEVER_CONNECTED,
682 .disconnect_timestamp = 0,
683 };
684 const struct zbus_channel *chan;
685 uint8_t msg_buf[MAX_MSG_SIZE];
686 int err;
687 int task_wdt_id;
688 const uint32_t wdt_timeout_ms =
689 CONFIG_LED_MODULE_WATCHDOG_TIMEOUT_SECONDS * MSEC_PER_SEC;
690 const uint32_t msg_processing_ms =
691 CONFIG_LED_MODULE_MSG_PROCESSING_TIME_SECONDS * MSEC_PER_SEC;
692 const k_timeout_t zbus_wait_ms = K_MSEC(wdt_timeout_ms - msg_processing_ms);
693
694 LOG_DBG("LED thread started - WDT: %d s, Processing: %d s, ZBUS Wait: %d s",
695 wdt_timeout_ms / MSEC_PER_SEC, msg_processing_ms / MSEC_PER_SEC,
696 (wdt_timeout_ms - msg_processing_ms) / MSEC_PER_SEC);
697
698 task_wdt_id = task_wdt_util_setup(wdt_timeout_ms, task_wdt_callback,
699 (void *)k_current_get());
700 if (task_wdt_id < 0) {
701 LOG_ERR("Failed to setup watchdog");
702 return;
703 }
704
705 /* Attach ZBUS subscriber to thread for priority handling */
706 zbus_obs_attach_to_thread(&led_subscriber);
707
708 /* Initialize acquiring timeout work item */
709 k_work_init_delayable(&state.acquiring_timeout, acquiring_timeout_handler);
710
711 /* Apply initial pattern (already set in current_status initialization) */
712 apply_led_pattern(&led_patterns[current_status]);
713
714 while (true) {
715 err = task_wdt_feed(task_wdt_id);
716 if (err) {
717 LOG_ERR("Failed to feed task watchdog, error: %d", err);
718 return;
719 }
720
721 /* Wait for message (MSG_SUBSCRIBER gets message copy directly) */
722 err = zbus_sub_wait_msg(&led_subscriber, &chan, msg_buf, zbus_wait_ms);
723 if (err == -ENOMSG) {
724 continue;
725 } else if (err) {
726 LOG_ERR("zbus_sub_wait_msg failed: %d", err);
727 return;
728 }
729
730 /* Process message based on channel */
731 if (chan == &NETWORK_CHAN) {
732 const struct network_msg *msg = MSG_TO_NETWORK_MSG_PTR(msg_buf);
733 process_network_msg(&state, msg);
734 } else if (chan == &CLOUD_CHAN) {
735 const struct cloud_msg *msg = MSG_TO_CLOUD_MSG_PTR(msg_buf);
736 process_cloud_msg(&state, msg);
737 } else if (chan == &FOTA_CHAN) {
738 const struct fota_msg *fota = MSG_TO_FOTA_MSG_PTR(msg_buf);
739 process_fota_msg(&state, fota);
740 } else if (chan == &TRIGGER_CHAN) {
741 const struct trigger_msg *trigger = MSG_TO_TRIGGER_MSG_PTR(msg_buf);
742 process_trigger_msg(&state, trigger);
743 } else {
744 LOG_WRN("Received message from unknown channel");
745 }
746
747 /* Update LED pattern based on new state */
748 update_led_pattern(&state);
749 }
750}
