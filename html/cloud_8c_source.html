161/* Cloud module states */
162enum cloud_module_state {
163 STATE_OPERATIONAL, /* Normal operation */
164 STATE_WAITING_FOR_MODEM, /* Waiting for modem initialization */
165 STATE_DISCONNECTED, /* Disconnected state */
166 STATE_CONNECTING, /* Connecting state */
167 STATE_CONNECTING_ATTEMPT, /* Attempting to connect to the cloud */
168 STATE_CLOUD_READY, /* Cloud ready state */
169};
170
171/* Construct state table */
172static const struct smf_state states[] = {
173 [STATE_OPERATIONAL] = SMF_CREATE_STATE(
174 state_operational_entry, state_operational_run, NULL,
175 NULL, /* No parent state */
176 &states[STATE_WAITING_FOR_MODEM]), /* Initial transition */
177 [STATE_WAITING_FOR_MODEM] = SMF_CREATE_STATE(
178 state_waiting_for_modem_entry, state_waiting_for_modem_run, NULL,
179 &states[STATE_OPERATIONAL], /* Parent state */
180 NULL), /* No initial transition */
181 [STATE_DISCONNECTED] = SMF_CREATE_STATE(
182 state_disconnected_entry, state_disconnected_run, NULL,
183 &states[STATE_OPERATIONAL], /* Parent state */
184 NULL), /* No initial transition */
185 [STATE_CONNECTING] = SMF_CREATE_STATE(
186 state_connecting_entry, state_connecting_run, state_connecting_exit,
187 &states[STATE_OPERATIONAL], /* Parent state */
188 &states[STATE_CONNECTING_ATTEMPT]), /* Initial transition */
189 [STATE_CONNECTING_ATTEMPT] =
190 SMF_CREATE_STATE(state_connecting_attempt_entry, NULL, NULL,
191 &states[STATE_CONNECTING], /* Parent state */
192 NULL), /* No initial transition */
193 [STATE_CLOUD_READY] =
194 SMF_CREATE_STATE(state_cloud_ready_entry, state_cloud_ready_run,
195 state_cloud_ready_exit,
196 &states[STATE_OPERATIONAL], /* Parent state */
197 NULL), /* No initial transition */
198};

367static void on_fota_downloaded(enum nrf_cloud_fota_type fota_type)
368{
369 LOG_INF("FOTA download completed - Type: %s", fota_type_to_str(fota_type));
370
371 /* Validate FOTA type - Phase 1 only supports APPLICATION */
372 if (fota_type != NRF_CLOUD_FOTA_APPLICATION) {
373 LOG_ERR("Unsupported FOTA type: %s (only APPLICATION supported in Phase 1)",
374 fota_type_to_str(fota_type));
375
376 /* Publish error status */
377 struct fota_msg fota_status = {
378 .status = FOTA_STATUS_ERROR,
379 .progress = 0,
380 };
381 zbus_chan_pub(&FOTA_CHAN, &fota_status, K_NO_WAIT);
382 return;
383 }
384
385 /* Publish final FOTA status for LED feedback */
386 struct fota_msg fota_status = {
387 .status = FOTA_STATUS_READY,
388 .progress = 100,
389 };
390 int err = zbus_chan_pub(&FOTA_CHAN, &fota_status, K_SECONDS(1));
391 if (err) {
392 LOG_ERR("Failed to publish FOTA ready status: %d", err);
393 }
394
395 /* Trigger graceful reboot via reboot module */
396 LOG_INF("Triggering graceful reboot for FOTA update");
397 reboot_request_publish(REBOOT_TYPE_FOTA);
398}

407static void fota_boot_confirm(void)
408{
409#if defined(CONFIG_BOOTLOADER_MCUBOOT)
410 static bool boot_confirmed = false;
411 int err;
412
413 if (boot_confirmed) {
414 return; /* Already confirmed */
415 }
416
417 /* Check if we're running a test image (just swapped from FOTA) */
418 if (!boot_is_img_confirmed()) {
419 LOG_INF("Confirming boot after FOTA update");
420
421 err = boot_write_img_confirmed();
422 if (err) {
423 LOG_ERR("Failed to confirm boot: %d - ROLLBACK RISK!", err);
424 /* Continue anyway - system is running, MCUboot will retry */
425 } else {
426 LOG_INF("Boot confirmed - FOTA update successful!");
427 boot_confirmed = true;
428 }
429 } else {
430 /* Not a test image, normal boot */
431 boot_confirmed = true;
432 LOG_DBG("Boot already confirmed (not FOTA update)");
433 }
434#endif /* CONFIG_BOOTLOADER_MCUBOOT */
435}

453/* Calculate backoff time for readiness check */
454static uint32_t calculate_backoff_time(uint32_t attempts)
455{
456 uint32_t backoff_time = CONFIG_APP_CLOUD_MODULE_READINESS_INITIAL_SECONDS;
457
458 if (IS_ENABLED(
459 CONFIG_APP_CLOUD_MODULE_READINESS_BACKOFF_TYPE_EXPONENTIAL)) {
460 backoff_time = CONFIG_APP_CLOUD_MODULE_READINESS_INITIAL_SECONDS
461 << (attempts - 1);
462 } else if (IS_ENABLED(
463 CONFIG_APP_CLOUD_MODULE_READINESS_BACKOFF_TYPE_LINEAR)) {
464 backoff_time =
465 CONFIG_APP_CLOUD_MODULE_READINESS_INITIAL_SECONDS +
466 ((attempts - 1) *
467 CONFIG_APP_CLOUD_MODULE_READINESS_BACKOFF_LINEAR_INCREMENT_SECONDS);
468 }
469
470 if (backoff_time > CONFIG_APP_CLOUD_MODULE_READINESS_BACKOFF_MAX_SECONDS) {
471 backoff_time = CONFIG_APP_CLOUD_MODULE_READINESS_BACKOFF_MAX_SECONDS;
472 }
473
474 return backoff_time;
475}

495/* nRF Cloud event handler */
496static void nrf_cloud_event_handler(const struct nrf_cloud_evt *nrf_cloud_evt)
497{
498 switch (nrf_cloud_evt->type) {
499 case NRF_CLOUD_EVT_TRANSPORT_CONNECTED:
500 LOG_DBG("NRF_CLOUD_EVT_TRANSPORT_CONNECTED");
501 break;
502 case NRF_CLOUD_EVT_TRANSPORT_CONNECTING:
503 LOG_DBG("NRF_CLOUD_EVT_TRANSPORT_CONNECTING");
504 break;
505 case NRF_CLOUD_EVT_USER_ASSOCIATION_REQUEST:
506 LOG_DBG("NRF_CLOUD_EVT_USER_ASSOCIATION_REQUEST");
507 LOG_ERR("Add the device to nRF Cloud to complete user "
508 "association, then restart the device");
509 break;
510 case NRF_CLOUD_EVT_USER_ASSOCIATED:
511 LOG_DBG("NRF_CLOUD_EVT_USER_ASSOCIATED");
512 break;
513 case NRF_CLOUD_EVT_READY:
514 LOG_INF("Cloud ready - device can send data");
515
516 /* Confirm boot after FOTA (if applicable) */
517 fota_boot_confirm();
518
519 cloud_status_publish(CLOUD_READY, K_SECONDS(1));
520 break;
521 case NRF_CLOUD_EVT_RX_DATA_GENERAL:
522 LOG_DBG("NRF_CLOUD_EVT_RX_DATA_GENERAL");
523 break;
524 case NRF_CLOUD_EVT_RX_DATA_DISCON:
525 LOG_DBG("NRF_CLOUD_EVT_RX_DATA_DISCON");
526 break;
527 case NRF_CLOUD_EVT_RX_DATA_LOCATION:
528 LOG_DBG("NRF_CLOUD_EVT_RX_DATA_LOCATION");
529 break;
530 case NRF_CLOUD_EVT_RX_DATA_SHADOW:
531 LOG_DBG("NRF_CLOUD_EVT_RX_DATA_SHADOW");
532 break;
533 case NRF_CLOUD_EVT_PINGRESP:
534 LOG_DBG("NRF_CLOUD_EVT_PINGRESP");
535 break;
536 case NRF_CLOUD_EVT_SENSOR_DATA_ACK:
537 LOG_DBG("NRF_CLOUD_EVT_SENSOR_DATA_ACK");
538 break;
539 case NRF_CLOUD_EVT_TRANSPORT_DISCONNECTED:
540 LOG_DBG("NRF_CLOUD_EVT_TRANSPORT_DISCONNECTED: %d",
541 nrf_cloud_evt->status);
542 private_cloud_status_publish(PRIV_CLOUD_DISCONNECTED, K_SECONDS(1));
543 break;
544 case NRF_CLOUD_EVT_FOTA_START:
545 LOG_INF("FOTA download started");
546 /* Publish FOTA status for LED feedback (initial slow blink) */
547 {
548 struct fota_msg fota_status = {
549 .status = FOTA_STATUS_DOWNLOADING,
550 .progress = 0,
551 };
552 zbus_chan_pub(&FOTA_CHAN, &fota_status, K_NO_WAIT);
553 }
554 break;
555 case NRF_CLOUD_EVT_FOTA_DONE: {
556 /* Extract FOTA type from event data */
557 enum nrf_cloud_fota_type fota_type = NRF_CLOUD_FOTA_TYPE__INVALID;
558
559 if (nrf_cloud_evt->data.ptr) {
560 fota_type = *((enum nrf_cloud_fota_type *)nrf_cloud_evt->data.ptr);
561 }
562
563 LOG_INF("NRF_CLOUD_EVT_FOTA_DONE - Type: %s",
564 fota_type_to_str(fota_type));
565
566 /* Handle download completion and trigger reboot */
567 on_fota_downloaded(fota_type);
568 break;
569 }
570 case NRF_CLOUD_EVT_FOTA_ERROR:
571 LOG_ERR("FOTA error: %d", nrf_cloud_evt->status);
572 /* Publish FOTA error status */
573 {
574 struct fota_msg fota_status = {
575 .status = FOTA_STATUS_ERROR,
576 .progress = 0,
577 };
578 zbus_chan_pub(&FOTA_CHAN, &fota_status, K_NO_WAIT);
579 }
580 break;
581 case NRF_CLOUD_EVT_TRANSPORT_CONNECT_ERROR:
582 LOG_DBG("NRF_CLOUD_EVT_TRANSPORT_CONNECT_ERROR: %d",
583 nrf_cloud_evt->status);
584 break;
585 case NRF_CLOUD_EVT_FOTA_JOB_AVAILABLE:
586 LOG_DBG("NRF_CLOUD_EVT_FOTA_JOB_AVAILABLE");
587 break;
588 case NRF_CLOUD_EVT_ERROR:
589 LOG_DBG("NRF_CLOUD_EVT_ERROR: %d", nrf_cloud_evt->status);
590 break;
591 default:
592 LOG_DBG("Unknown event type: %d", nrf_cloud_evt->type);
593 break;
594 }
595}

873/*
874 * =================================================================================
875 * Thread
876 * =================================================================================
877 *
878 * Defines the cloud module's thread and its main loop, which manages the watchdog,
879 * waits for ZBUS messages, and runs the state machine.
880 */
881
882/* Cloud module thread */
883static void cloud_thread(void)
884{
885 int err;
886 int task_wdt_id;
887 const uint32_t wdt_timeout_ms =
888 (CONFIG_APP_CLOUD_MODULE_WATCHDOG_TIMEOUT_SECONDS * MSEC_PER_SEC);
889 const uint32_t msg_processing_ms =
890 (CONFIG_APP_CLOUD_MODULE_MSG_PROCESSING_TIME_SECONDS *
891 MSEC_PER_SEC);
892 const k_timeout_t zbus_wait_ms = K_MSEC(wdt_timeout_ms - msg_processing_ms);
893 struct cloud_state_object cloud_state_object = { 0 };
894
895 LOG_DBG("Cloud module started - WDT: %d s, Processing: %d s, ZBUS Wait: %d s",
896 wdt_timeout_ms / MSEC_PER_SEC, msg_processing_ms / MSEC_PER_SEC,
897 (wdt_timeout_ms - msg_processing_ms) / MSEC_PER_SEC);
898
899 task_wdt_id = task_wdt_util_setup(wdt_timeout_ms, task_wdt_callback,
900 (void *)k_current_get());
901 if (task_wdt_id < 0) {
902 /* todo: reboot_request_publish(REBOOT_TYPE_ERROR); */
903 return;
904 }
905
906 zbus_obs_attach_to_thread(&cloud_subscriber);
907
908 smf_set_initial(SMF_CTX(&cloud_state_object), &states[STATE_OPERATIONAL]);
909
910 /*
911 * The main loop of the cloud module thread:
912 * - Feeds the watchdog
913 * - Waits for ZBUS messages with a timeout
914 * - Runs the state machine for each received message
915 * - Handles errors and continues operation unless a critical failure occurs
916 */
917 while (true) {
918 err = task_wdt_feed(task_wdt_id);
919 if (err) {
920 LOG_ERR("Failed to feed task watchdog, error: %d", err);
921 /* todo: reboot_request_publish(REBOOT_TYPE_ERROR); */
922 return;
923 }
924
925 err = zbus_sub_wait_msg(&cloud_subscriber, &cloud_state_object.chan,
926 cloud_state_object.msg_buf, zbus_wait_ms);
927 if (err == -ENOMSG) {
928 continue;
929 } else if (err) {
930 LOG_ERR("zbus_sub_wait_msg, error: %d", err);
931 /* todo: reboot_request_publish(REBOOT_TYPE_ERROR); */
932 return;
933 }
934
935 err = smf_run_state(SMF_CTX(&cloud_state_object));
936 if (err) {
937 LOG_ERR("smf_run_state, error: %d", err);
938 /* todo: reboot_request_publish(REBOOT_TYPE_ERROR); */
939 return;
940 }
941 }
942}
