186/* Re-sync interval: 6 hours to compensate for clock drift (RC oscillator Â±250 ppm) */
187#define TIME_RESYNC_INTERVAL_MS (6 * 60 * 60 * 1000)
188
189/* Time sync state - tracks if we've ever had network connection to obtain unix time */
190static struct {
191 bool time_synced; /* Have we synchronized with unix time? */
192 int64_t unix_epoch_offset_ms; /* Unix epoch at boot = unix_time - uptime */
193 int64_t last_sync_uptime_ms; /* Uptime when last sync occurred (for drift compensation) */
194} time_sync_state = {
195 .time_synced = false,
196 .unix_epoch_offset_ms = 0,
197 .last_sync_uptime_ms = 0,
198};

206static int64_t uptime_to_unix_epoch(int64_t uptime_ms)
207{
208 if (!time_sync_state.time_synced) {
209 return 0; /* No time sync yet */
210 }
211
212 return time_sync_state.unix_epoch_offset_ms + uptime_ms;
213}

226static void sync_time(int64_t current_uptime_ms, bool force)
227{
228 int64_t unix_time_ms;
229 int err;
230
231 if (time_sync_state.time_synced && !force) {
232 LOG_DBG("Time already synced, ignoring");
233 return;
234 }
235
236 /* Get current unix time from date_time library (updated via nRF Cloud) */
237 err = date_time_now(&unix_time_ms);
238 if (err) {
239 LOG_ERR("Failed to get unix time for sync: %d", err);
240 return;
241 }
242
243 /* Calculate offset: unix_epoch_at_boot = unix_time_now - uptime_now */
244 time_sync_state.unix_epoch_offset_ms = unix_time_ms - current_uptime_ms;
245 time_sync_state.time_synced = true;
246 time_sync_state.last_sync_uptime_ms = current_uptime_ms;
247
248 LOG_INF("Time %s! Unix epoch offset: %lld ms",
249 force ? "re-synchronized" : "synchronized",
250 time_sync_state.unix_epoch_offset_ms);
251 LOG_INF("Current uptime: %lld ms -> Unix time: %lld ms", current_uptime_ms,
252 unix_time_ms);
253}
254
260static bool time_needs_resync(void)
261{
262 if (!time_sync_state.time_synced) {
263 return false; /* Not yet synced */
264 }
265
266 int64_t time_since_last_sync =
267 k_uptime_get() - time_sync_state.last_sync_uptime_ms;
268
269 return time_since_last_sync > TIME_RESYNC_INTERVAL_MS;
270}

280static void date_time_evt_handler(const struct date_time_evt *evt)
281{
282 /* Only sync on NTP events - ignore modem/external sources */
283 if (evt->type != DATE_TIME_OBTAINED_NTP) {
284 return;
285 }
286
287 LOG_INF("DATE_TIME_OBTAINED_NTP event received, syncing time");
288
289 int64_t unix_time_ms;
290 int err = date_time_now(&unix_time_ms);
291 if (err) {
292 LOG_ERR("Failed to get current time: %d", err);
293 return;
294 }
295
296 sync_time(k_uptime_get(), false);
297}

370static void combine_and_queue_data(void)
371{
372 int err;
373 struct combined_energy_msg combined = { 0 };
374
375 /* Stop collection timer */
376 k_timer_stop(&collection_timer);
377
378 /* Uptime timestamp of this combine event */
379 combined.uptime_ms = k_uptime_get();
380
381 /* Add P1 data if collected */
382 if (collected_p1.has_data) {
383 combined.has_p1_data = true;
384 combined.p1_data = collected_p1.data;
385 LOG_DBG("Adding P1 data to combined message");
386 } else {
387 combined.has_p1_data = false;
388 LOG_WRN("No P1 data received during collection window");
389 }
390
391 /* Add Pulse data if collected */
392 if (collected_pulse.has_data) {
393 combined.has_pulse_data = true;
394 combined.pulse_data = collected_pulse.data;
395 LOG_DBG("Adding Pulse data to combined message");
396 } else {
397 combined.has_pulse_data = false;
398 LOG_WRN("No Pulse data received during collection window");
399 }
400
401 /* Check if we have ANY data to queue */
402 if (!collected_p1.has_data && !collected_pulse.has_data) {
403 LOG_ERR("No data collected in window - skipping queue (P1 and Pulse both failed)");
404 return; /* Don't queue empty messages - buffers will be cleared on next window start */
405 }
406
407 /* Queue the combined message */
408 err = queue_combined_message(&combined);
409 if (err) {
410 LOG_ERR("Failed to queue combined message: %d", err);
411 return;
412 }
413
414 /* Log what we collected */
415 if (collected_p1.has_data && collected_pulse.has_data) {
416 LOG_INF("Queued complete message (P1 + Pulse)");
417 } else if (collected_p1.has_data) {
418 LOG_WRN("Queued incomplete message (P1 only)");
419 } else {
420 /* Must be Pulse only - we already checked that at least one has data */
421 LOG_WRN("Queued incomplete message (Pulse only)");
422 }
423
424 /* Clear collection buffers */
425 collected_p1.has_data = false;
426 collected_pulse.has_data = false;
427}

503static int queue_combined_message(const struct combined_energy_msg *msg)
504{
505 int err;
506
507 if (!msg) {
508 LOG_ERR("Invalid message pointer");
509 return -EINVAL;
510 }
511
512 /* Check if queue is full */
513 if (k_msgq_num_free_get(&combined_energy_msgq) == 0) {
514 struct combined_energy_msg dropped;
515
516 /* Drop oldest message (FIFO) */
517 err = k_msgq_get(&combined_energy_msgq, &dropped, K_NO_WAIT);
518 if (err == 0) {
519 LOG_WRN("Queue full, dropped oldest message (timestamp: %lld)",
520 dropped.uptime_ms);
521 } else {
522 LOG_ERR("Failed to drop oldest message: %d", err);
523 }
524 }
525
526 /* Add new message to queue */
527 err = k_msgq_put(&combined_energy_msgq, msg, K_NO_WAIT);
528 if (err) {
529 LOG_ERR("Failed to queue message: %d", err);
530 return err;
531 }
532
533 /* Log queue statistics */
534 uint32_t used = k_msgq_num_used_get(&combined_energy_msgq);
535 uint32_t free = k_msgq_num_free_get(&combined_energy_msgq);
536 LOG_INF("Queued combined message (used: %u, free: %u)", used, free);
537
538 return 0;
539}

562static int build_combined_energy_json(const struct combined_energy_msg *msg,
563 struct nrf_cloud_obj *const obj)
564{
565 int err;
566
567 if (!msg || !obj) {
568 return -EINVAL;
569 }
570
571 /* Initialize as nRF Cloud device message with board-specific appId */
572 err = nrf_cloud_obj_msg_init(obj, CONFIG_NRF_CLOUD_ENERGY_DATA_MSG_APPID,
573 "DATA");
574 if (err) {
575 LOG_ERR("Failed to init cloud message: %d", err);
576 return err;
577 }
578
579 /* Add data format version identifier */
580 err = nrf_cloud_obj_str_add(obj, "dataFormatVersion",
581 CONFIG_DATA_FORMAT_VERSION_STRING, false);
582 if (err) {
583 LOG_ERR("Failed to add data format version: %d", err);
584 return err;
585 }
586
587 /* Convert uptime to Unix epoch timestamp for message (if synced) */
588 int64_t timestamp_epoch_ms = uptime_to_unix_epoch(msg->uptime_ms);
589
590 /* Add timestamp (Unix epoch if synced, uptime otherwise) */
591 err = nrf_cloud_obj_ts_add(obj, timestamp_epoch_ms);
592 if (err) {
593 LOG_ERR("Failed to add timestamp: %d", err);
594 return err;
595 }
596
597 /* Add time sync status flag (minimal JSON: "ts_sync":1 or "ts_sync":0) */
598 err = nrf_cloud_obj_bool_add(obj, "ts_sync", time_sync_state.time_synced,
599 false);
600 if (err) {
601 LOG_ERR("Failed to add ts_sync flag: %d", err);
602 return err;
603 }
604
605 if (!time_sync_state.time_synced) {
606 LOG_WRN("Time not synced, sending uptime as timestamp");
607 }
608
609 /* Create nested "data" object for sensor readings */
610 NRF_CLOUD_OBJ_JSON_DEFINE(data_obj);
611
612 err = nrf_cloud_obj_init(&data_obj);
613 if (err) {
614 LOG_ERR("Failed to init data object: %d", err);
615 return err;
616 }
617
618 /* Add P1 data if available */
619 if (msg->has_p1_data) {
620 NRF_CLOUD_OBJ_JSON_DEFINE(p1_obj);
621
622 err = nrf_cloud_obj_init(&p1_obj);
623 if (err) {
624 LOG_ERR("Failed to init P1 object: %d", err);
625 nrf_cloud_obj_free(&data_obj);
626 return err;
627 }
628
629 const struct p_one_msg *p1 = &msg->p1_data;
630
631 /* P1 timestamps are already Unix epoch from meter (OBIS 0-0:1.0.0) - no conversion needed! */
632 int64_t p1_timestamp_epoch_ms = p1->timestamp_epoch_ms;
633
634 /* Add P1 timestamp */
635 err = nrf_cloud_obj_num_add(&p1_obj, "ts",
636 (double)p1_timestamp_epoch_ms, false);
637 if (err) {
638 LOG_ERR("Failed to add P1 timestamp: %d", err);
639 goto cleanup_p1;
640 }
641
642 /* Energy consumed (Wh) */
643 err = nrf_cloud_obj_num_add(
644 &p1_obj, "ec1", (double)p1->energy_consumed_wh_tariff1, false);
645 if (err)
646 goto cleanup_p1;
647
648 err = nrf_cloud_obj_num_add(
649 &p1_obj, "ec2", (double)p1->energy_consumed_wh_tariff2, false);
650 if (err)
651 goto cleanup_p1;
652
653 /* Energy delivered (Wh) */
654 err = nrf_cloud_obj_num_add(
655 &p1_obj, "ed1", (double)p1->energy_delivered_wh_tariff1, false);
656 if (err)
657 goto cleanup_p1;
658
659 err = nrf_cloud_obj_num_add(
660 &p1_obj, "ed2", (double)p1->energy_delivered_wh_tariff2, false);
661 if (err)
662 goto cleanup_p1;
663
664 /* Gas consumed (dmÂ³ = liters) */
665 err = nrf_cloud_obj_num_add(&p1_obj, "gdm3",
666 (double)p1->gas_consumed_dm3, false);
667 if (err)
668 goto cleanup_p1;
669
670 /* Power consumed statistics (W) */
671 err = nrf_cloud_obj_num_add(&p1_obj, "pca",
672 (double)p1->power_consumed_avg_w, false);
673 if (err)
674 goto cleanup_p1;
675
676 err = nrf_cloud_obj_num_add(&p1_obj, "pcx",
677 (double)p1->power_consumed_max_w, false);
678 if (err)
679 goto cleanup_p1;
680
681 /* P1 peak timestamps are already epoch (from meter OBIS 0-0:1.0.0) */
682 err = nrf_cloud_obj_num_add(
683 &p1_obj, "pcxts",
684 (double)p1->power_consumed_max_timestamp_epoch_ms, false);
685 if (err)
686 goto cleanup_p1;
687
688 err = nrf_cloud_obj_num_add(&p1_obj, "pcn",
689 (double)p1->power_consumed_min_w, false);
690 if (err)
691 goto cleanup_p1;
692
693 /* Power delivered statistics (W) */
694 err = nrf_cloud_obj_num_add(&p1_obj, "pda",
695 (double)p1->power_delivered_avg_w, false);
696 if (err)
697 goto cleanup_p1;
698
699 err = nrf_cloud_obj_num_add(&p1_obj, "pdx",
700 (double)p1->power_delivered_max_w, false);
701 if (err)
702 goto cleanup_p1;
703
704 /* P1 peak timestamps are already epoch (from meter OBIS 0-0:1.0.0) */
705 err = nrf_cloud_obj_num_add(
706 &p1_obj, "pdxts",
707 (double)p1->power_delivered_max_timestamp_epoch_ms, false);
708 if (err)
709 goto cleanup_p1;
710
711 err = nrf_cloud_obj_num_add(&p1_obj, "pdn",
712 (double)p1->power_delivered_min_w, false);
713 if (err)
714 goto cleanup_p1;
715
716 /* Add P1 object to data object */
717 err = nrf_cloud_obj_object_add(&data_obj, "p1", &p1_obj, false);
718 if (err) {
719 LOG_ERR("Failed to add P1 to data: %d", err);
720 goto cleanup_p1;
721 }
722
723 /* p1_obj is now managed by data_obj, don't free it */
724 goto p1_done;
725
726cleanup_p1:
727 nrf_cloud_obj_free(&p1_obj);
728 nrf_cloud_obj_free(&data_obj);
729 return err;
730
731p1_done:
732 LOG_DBG("P1 data added to message");
733 }
734
735 /* Add Pulse data if available */
736 if (msg->has_pulse_data) {
737 NRF_CLOUD_OBJ_JSON_DEFINE(pulse_obj);
738
739 err = nrf_cloud_obj_init(&pulse_obj);
740 if (err) {
741 LOG_ERR("Failed to init Pulse object: %d", err);
742 nrf_cloud_obj_free(&data_obj);
743 return err;
744 }
745
746 const struct energy_pulse_msg *pulse = &msg->pulse_data;
747
748 /* Convert Pulse uptime to Unix epoch */
749 int64_t pulse_timestamp_epoch_ms =
750 uptime_to_unix_epoch(pulse->uptime_ms);
751
752 /* Add Pulse timestamp */
753 err = nrf_cloud_obj_num_add(&pulse_obj, "ts",
754 (double)pulse_timestamp_epoch_ms, false);
755 if (err) {
756 LOG_ERR("Failed to add Pulse timestamp: %d", err);
757 goto cleanup_pulse;
758 }
759
760 /* Add pulse counters */
761 err = nrf_cloud_obj_num_add(&pulse_obj, "ch1",
762 (double)pulse->pulse_energy_ch1_wh, false);
763 if (err)
764 goto cleanup_pulse;
765
766 err = nrf_cloud_obj_num_add(&pulse_obj, "ch2",
767 (double)pulse->pulse_energy_ch2_wh, false);
768 if (err)
769 goto cleanup_pulse;
770
771 /* Add Pulse object to data object */
772 err = nrf_cloud_obj_object_add(&data_obj, "pulse", &pulse_obj, false);
773 if (err) {
774 LOG_ERR("Failed to add Pulse to data: %d", err);
775 goto cleanup_pulse;
776 }
777
778 /* pulse_obj is now managed by data_obj */
779 goto pulse_done;
780
781cleanup_pulse:
782 nrf_cloud_obj_free(&pulse_obj);
783 nrf_cloud_obj_free(&data_obj);
784 return err;
785
786pulse_done:
787 LOG_DBG("Pulse data added to message");
788 }
789
790 /* Add the complete data object to the message */
791 err = nrf_cloud_obj_object_add(obj, "data", &data_obj, false);
792 if (err) {
793 LOG_ERR("Failed to add data object to message: %d", err);
794 nrf_cloud_obj_free(&data_obj);
795 return err;
796 }
797
798 LOG_INF("Combined energy JSON built successfully (P1: %s, Pulse: %s)",
799 msg->has_p1_data ? "YES" : "NO",
800 msg->has_pulse_data ? "YES" : "NO");
801
802 return 0;
803}

835 (struct message_queue_state_object *)o;
836
837 /* Handle cloud status updates - track but don't change state */
838 if (state_object->chan == &CLOUD_CHAN) {
839 const struct cloud_msg *msg =
840 MSG_TO_CLOUD_MSG_PTR(state_object->msg_buf);
841 update_cloud_status(msg);
842 smf_set_handled(SMF_CTX(state_object));
843 return;
844 }
845
846 /* Handle network status for periodic time re-sync (drift compensation) */
847 if (state_object->chan == &NETWORK_CHAN) {
848 const struct network_msg *msg =
849 MSG_TO_NETWORK_MSG_PTR(state_object->msg_buf);
850
851 /* Check if periodic re-sync is needed (drift compensation) */
852 if (network_is_connected(msg) && time_needs_resync()) {
853 int64_t current_uptime = k_uptime_get();
854 sync_time(current_uptime, true); /* Force re-sync */
855 }
856
857 smf_set_handled(SMF_CTX(state_object));
858 return;
859 }
860
861 /* Handle publish request - ALWAYS collect data regardless of cloud status */
862 if (state_object->chan == &DATA_PUBLISH_REQUEST_CHAN) {
863 LOG_INF("Publish request received, starting data collection");
864 smf_set_state(SMF_CTX(state_object), &states[STATE_COLLECTING_DATA]);
865 return;
866 }
867
868 /* Handle send request - only if cloud is ready */
869 if (state_object->chan == &DATA_SEND_REQUEST_CHAN) {
870 if (is_cloud_ready) {
871 smf_set_state(SMF_CTX(state_object),
872 &states[STATE_SEND_MSG_ATTEMPT]);
873 } else {
874 LOG_WRN("Send request ignored - cloud not ready (queue size: %u)",
875 get_queue_usage());
876 }
877 smf_set_handled(SMF_CTX(state_object));
878 return;
879 }
880
881 /* Ignore P1 and Pulse data outside collection window */
882 if (state_object->chan == &P_ONE_CHAN ||
883 state_object->chan == &ENERGY_PULSE_CHAN) {
884 LOG_WRN("Data received outside collection window, ignoring");
885 smf_set_handled(SMF_CTX(state_object));
886 return;
887 }

1005static void state_send_msg_attempt_entry(void *o)
1006{
1007 struct message_queue_state_object *state_object =
1008 (struct message_queue_state_object *)o;
1009
1010 /* Double-check cloud readiness (defense in depth) */
1011 if (!is_cloud_ready) {
1012 LOG_WRN("Cloud no longer ready, aborting send");
1013 smf_set_state(SMF_CTX(state_object), &states[STATE_OPERATIONAL]);
1014 return;
1015 }
1016
1017 uint32_t queue_size = k_msgq_num_used_get(&combined_energy_msgq);
1018
1019 if (queue_size == 0) {
1020 LOG_DBG("Queue empty, nothing to send");
1021 smf_set_state(SMF_CTX(state_object), &states[STATE_OPERATIONAL]);
1022 return;
1023 }
1024
1025 LOG_INF("Attempting to send %u queued messages", queue_size);
1026
1027 /* Limit messages per cycle to prevent blocking too long */
1028#define MAX_MSGS_PER_CYCLE 10
1029
1030 int sent_count = 0;
1031
1032 /* Send messages until queue empty or limit reached */
1033 while (k_msgq_num_used_get(&combined_energy_msgq) > 0 &&
1034 sent_count < MAX_MSGS_PER_CYCLE) {
1035 struct combined_energy_msg msg;
1036
1037 /* CRITICAL: Use peek-then-get pattern to prevent data loss
1038 * - Peek keeps message in queue during send attempt
1039 * - Only remove after confirmed successful transmission
1040 * - This ensures no data loss on network failures
1041 */
1042 int err = k_msgq_peek(&combined_energy_msgq, &msg);
1043
1044 if (err != 0) {
1045 LOG_ERR("Failed to peek message from queue: %d", err);
1046 break;
1047 }
1048
1049 /* Build JSON object */
1050 NRF_CLOUD_OBJ_JSON_DEFINE(cloud_obj);
1051
1052 err = build_combined_energy_json(&msg, &cloud_obj);
1053 if (err) {
1054 LOG_ERR("Failed to build JSON for message %d: %d", sent_count + 1,
1055 err);
1056 /* Drop corrupt message - remove from queue and continue */
1057 k_msgq_get(&combined_energy_msgq, &msg, K_NO_WAIT);
1058 continue;
1059 }
1060
1061 /* Send to nRF Cloud - SDK handles encoding internally */
1062 struct nrf_cloud_tx_data tx_data = {
1063 .obj = &cloud_obj,
1064 .topic_type = NRF_CLOUD_TOPIC_MESSAGE,
1065 .qos = MQTT_QOS_1_AT_LEAST_ONCE,
1066 };
1067
1068 err = nrf_cloud_send(&tx_data);
1069
1070 /* Cleanup cloud object */
1071 nrf_cloud_obj_free(&cloud_obj);
1072
1073 if (err) {
1074 LOG_ERR("Failed to send message %d to cloud: %d", sent_count + 1,
1075 err);
1076 /* On send failure, MESSAGE STAYS IN QUEUE for retry */
1077 /* Break to retry entire queue in next cycle */
1078 break;
1079 }
1080
1081 /* SUCCESS: Now safe to remove message from queue */
1082 k_msgq_get(&combined_energy_msgq, &msg, K_NO_WAIT);
1083
1084 sent_count++;
1085 LOG_INF("Message %d/%d sent successfully (ts: %lld)", sent_count,
1086 queue_size, msg.uptime_ms);
1087 }
1088
1089 uint32_t remaining = k_msgq_num_used_get(&combined_energy_msgq);
1090 LOG_INF("Sent %d messages, %u remaining in queue", sent_count, remaining);
1091
1092 /* Always return to OPERATIONAL - never block on cloud! */
1093 smf_set_state(SMF_CTX(state_object), &states[STATE_OPERATIONAL]);
1094}

1108static void message_queue_thread(void)
1109{

1111 int task_wdt_id;
1112 const uint32_t wdt_timeout_ms =
1113 (CONFIG_APP_MESSAGE_QUEUE_MODULE_WATCHDOG_TIMEOUT_SECONDS *
1114 MSEC_PER_SEC);
1115 const uint32_t msg_processing_ms =
1116 (CONFIG_APP_MESSAGE_QUEUE_MODULE_MSG_PROCESSING_TIME_SECONDS *
1117 MSEC_PER_SEC);
1118 const k_timeout_t zbus_wait_ms = K_MSEC(wdt_timeout_ms - msg_processing_ms);
1119 struct message_queue_state_object message_queue_state_object = { 0 };
1120
1121 LOG_DBG("Message queue module started - WDT: %d s, Processing: %d s, ZBUS Wait: %d "
1122 "s",
1123 wdt_timeout_ms / MSEC_PER_SEC, msg_processing_ms / MSEC_PER_SEC,
1124 (wdt_timeout_ms - msg_processing_ms) / MSEC_PER_SEC);
1125
1126 task_wdt_id = task_wdt_util_setup(wdt_timeout_ms, task_wdt_callback,
1127 (void *)k_current_get());
1128 if (task_wdt_id < 0) {
1129 /* Task watchdog setup failure is fatal */
1130 reboot_request_publish(REBOOT_TYPE_ERROR);
1131 return;
1132 }
1133
1134 /* Register date_time event handler for NTP sync notifications */
1135 date_time_register_handler(date_time_evt_handler);
1136
1137 zbus_obs_attach_to_thread(&message_queue_subscriber);
1138
1139 smf_set_initial(SMF_CTX(&message_queue_state_object),
1140 &states[STATE_OPERATIONAL]);
1141
1142 while (true) {
1143 err = task_wdt_feed(task_wdt_id);
1144 if (err) {
1145 LOG_ERR("Failed to feed task watchdog, error: %d", err);
1146
1147 /* Watchdog feed failure is fatal */
1148 reboot_request_publish(REBOOT_TYPE_ERROR);
1149 return;
1150 }
1151
1152 err = zbus_sub_wait_msg(&message_queue_subscriber,
1153 &message_queue_state_object.chan,
1154 message_queue_state_object.msg_buf,
1155 zbus_wait_ms);
1156 if (err == -ENOMSG) {
1157 continue;
1158 } else if (err) {
1159 LOG_ERR("zbus_sub_wait_msg, error: %d", err);
1160
1161 /* ZBUS message wait failure is fatal */
1162 reboot_request_publish(REBOOT_TYPE_ERROR);
1163 return;
1164 }
1165
1166 err = smf_run_state(SMF_CTX(&message_queue_state_object));
1167 if (err) {
1168 LOG_ERR("smf_run_state, error: %d", err);
1169
1170 /* State machine failure is fatal */
1171 reboot_request_publish(REBOOT_TYPE_ERROR);
1172 return;
1173 }
1174 }
1175}
