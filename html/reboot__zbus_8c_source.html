63void reboot_request_publish(const enum reboot_type type)
64{
65 const struct reboot_msg msg = {
66 .type = REBOOT_REQ,
67 .reboot_type = type,
68 };
69 int err;
70
71 /* Use 5s timeout: VDED should complete in milliseconds. If it takes longer,
72 * there's a bug (blocking listener callback or deadlock) that should be detected.
73 */
74 err = zbus_chan_pub(&REBOOT_CHAN, &msg, K_SECONDS(5));
75 if (err) {
76 LOG_ERR("Failed to publish reboot request: %d", err);
77 ZBUS_REBOOT_CHAN_FATAL_ERROR_HANDLE();
78 }
79
80 LOG_INF("Reboot request published, type: %s", get_reboot_type_str(type));
81}
