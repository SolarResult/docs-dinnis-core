4 * Timing Module
5 *
6 * Centralized timing module that manages periodic events for the system.
7 * Publishes three types of timing events on ZBUS channels:
8 * - DATA_ACQUISITION_TRIGGER_CHAN: Fast triggers for data acquisition (e.g., P1 UART reading)
9 * - DATA_PUBLISH_REQUEST_CHAN: Slow triggers for data publishing (e.g., P1 stats, Pulse counters)
10 * - DATA_SEND_REQUEST_CHAN: Send triggers for cloud data transmission
11 *
12 * The publish request includes a configurable offset to avoid collision with the last
13 * acquisition before publishing.
14 */
15

146static void fast_acquisition_work_fn(struct k_work *work)
147{
148 ARG_UNUSED(work);
149
150 struct timing_event_msg msg = {
151 .timestamp = k_uptime_get(),
152 };
153
154 int err = zbus_chan_pub(&DATA_ACQUISITION_TRIGGER_CHAN, &msg, K_SECONDS(1));
155
156 if (err) {
157 LOG_ERR("Failed to publish acquisition trigger: %d", err);
158 } else {
159 LOG_DBG("Published acquisition trigger (ts: %lld)", msg.timestamp);
160 }
161}

225static int timing_module_init(void)
226{
227 LOG_INF("Initializing Timing module");
228
229 /*
230 * Calculate slow timer parameters with offset.
231 * First trigger: SLOW_INTERVAL - OFFSET (e.g., 585s for 600s interval with 15s offset)
232 * Period: SLOW_INTERVAL (e.g., 600s)
233 * This ensures publish happens before next acquisition cycle.
234 */
235 uint32_t slow_first_delay_ms =
236 (CONFIG_APP_TIMING_MODULE_SLOW_INTERVAL_SECONDS -
237 CONFIG_APP_TIMING_MODULE_PUBLISH_OFFSET_SECONDS) *
238 1000;
239 uint32_t slow_period_ms =
240 CONFIG_APP_TIMING_MODULE_SLOW_INTERVAL_SECONDS * 1000;
241
242 /*
243 * Start periodic timers.
244 * Note: k_timer_start() with non-zero period creates an infinite periodic timer.
245 * From Zephyr docs: "If the timer's period is zero, the timer enters the stopped
246 * state; otherwise, the timer restarts with a new duration equal to its period."
247 * This means our timers will run forever until explicitly stopped with k_timer_stop().
248 */
249 k_timer_start(&fast_acquisition_timer,
250 K_SECONDS(CONFIG_APP_TIMING_MODULE_FAST_INTERVAL_SECONDS),
251 K_SECONDS(CONFIG_APP_TIMING_MODULE_FAST_INTERVAL_SECONDS));
252
253 k_timer_start(&slow_publish_timer, K_MSEC(slow_first_delay_ms),
254 K_MSEC(slow_period_ms));
255
256 k_timer_start(&send_timer,
257 K_SECONDS(CONFIG_APP_TIMING_MODULE_SEND_INTERVAL_SECONDS),
258 K_SECONDS(CONFIG_APP_TIMING_MODULE_SEND_INTERVAL_SECONDS));
259
260 LOG_INF("Timing module started:");
261 LOG_INF(" Fast acquisition: every %d seconds",
262 CONFIG_APP_TIMING_MODULE_FAST_INTERVAL_SECONDS);
263 LOG_INF(" Slow publish: first at %d seconds, then every %d seconds (offset %d)",
264 slow_first_delay_ms / 1000,
265 CONFIG_APP_TIMING_MODULE_SLOW_INTERVAL_SECONDS,
266 CONFIG_APP_TIMING_MODULE_PUBLISH_OFFSET_SECONDS);
267 LOG_INF(" Send: every %d seconds",
268 CONFIG_APP_TIMING_MODULE_SEND_INTERVAL_SECONDS);
269
270 return 0;
271}
